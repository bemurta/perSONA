92 void CVAPortaudioBackend::threadInitialize()
93 {
94  ITAPortaudioInterface::ITA_PA_ERRORCODE e;
95 
96  int iDriverID = -1;
97  try
98  {
99  iDriverID = StringToInt( m_pConfig->sDevice );
100  }
101  catch( ITAException& e )
102  {
103  if( e.iErrorCode != ITAException::PARSE_ERROR )
104  throw e;
105  }
106 
107  if( iDriverID == -1 )
108  {
109  if( ( e = m_pITAPA->Initialize() ) != ITAPortaudioInterface::ITA_PA_NO_ERROR )
110  ITA_EXCEPT1( INVALID_PARAMETER, ITAPortaudioInterface::GetErrorCodeString( e ).c_str() );
111  iDriverID = m_pITAPA->GetDefaultOutputDevice(); // Schneller, wenn Portaudio initialisiert ist
112  }
113  else
114  {
115  if( ( e = m_pITAPA->Initialize( iDriverID ) ) != ITAPortaudioInterface::ITA_PA_NO_ERROR )
116  ITA_EXCEPT1( INVALID_PARAMETER, ITAPortaudioInterface::GetErrorCodeString( e ).c_str() );
117  }
118 
119  // Anzahl Kanäle in die Config übernehmen
120  m_pConfig->iInputChannels = m_pITAPA->GetNumInputChannels( iDriverID );
121  m_pConfig->iOutputChannels = m_pITAPA->GetNumOutputChannels( iDriverID );
122 
123  m_pITAPA->SetRecordEnabled( m_pConfig->iInputChannels > 0 );
124  m_pITAPA->SetPlaybackEnabled( m_pConfig->iOutputChannels > 0 );
125 
126  double dSampleRate;
127  m_pITAPA->GetDriverSampleRate( iDriverID, dSampleRate );
128 
129  if( m_pConfig->dSampleRate != dSampleRate )
130  ITA_EXCEPT1( INVALID_PARAMETER, "Samplerate " + DoubleToString( m_pConfig->dSampleRate ) + " is invalid for this audio device." );
131 
132  m_oOutputStreamProps.uiBlocklength = ( unsigned int ) m_pConfig->iBuffersize;
133  m_oOutputStreamProps.uiChannels = ( unsigned int ) m_pITAPA->GetNumOutputChannels( iDriverID );
134 }
