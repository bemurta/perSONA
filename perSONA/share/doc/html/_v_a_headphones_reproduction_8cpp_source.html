53 CVAHeadphonesReproduction::CVAHeadphonesReproduction( const CVAAudioReproductionModuleInitParams& oParams )
54 : CVAObject( oParams.sClass + ":" + oParams.sID )
55 , m_oParams( oParams )
56 , m_pHPEQStreamFilter( NULL )
57 {
58  m_oParams.pCore->RegisterModule( this );
59 
60  CVAConfigInterpreter conf( *(m_oParams.pConfig) );
61 
62  // Gains
63  if( oParams.pConfig->HasKey( "HpIRInvCalibrationGain" ) && oParams.pConfig->HasKey( "HpIRInvCalibrationGainDecibel" ) )
64  VA_EXCEPT2( INVALID_PARAMETER, "HPIRInvCalibrationGain was included in the configuration both as factor and decibel and I can't decide which one is correct." );
65 
66  double dEqualizationCalibrationGain;
67  if( oParams.pConfig->HasKey( "HpIRInvCalibrationGainDecibel" ) )
68  {
69  double dEqualizationCalibrationGainDecibel;
70  conf.ReqNumber( "HpIRInvCalibrationGainDecibel", dEqualizationCalibrationGainDecibel );
71  dEqualizationCalibrationGain = db20_to_ratio( dEqualizationCalibrationGainDecibel );
72  }
73  else if( oParams.pConfig->HasKey( "HpIRInvCalibrationGain" ) )
74  {
75  conf.ReqNumber( "HpIRInvCalibrationGain", dEqualizationCalibrationGain );
76  }
77  else
78  {
79  dEqualizationCalibrationGain = 1.0f;
80  }
81 
82  // Inverse IR file
83  std::string sFilePathRaw;
84  conf.ReqString( "HpIRInvFile", sFilePathRaw );
85 
86  std::string sFilePath = m_oParams.pCore->GetCoreConfig()->mMacros.SubstituteMacros( sFilePathRaw );
87 
88  m_sfHpIRInv.Load( sFilePath ); // sets length to given number of samples
89 
90  if( m_sfHpIRInv.channels() != 2 )
91  VA_EXCEPT2( INVALID_PARAMETER, "Headphone EQ impulse response '" + sFilePath + "' requires two channels" );
92 
93  int iHpIRInvFilterLength;
94  conf.OptInteger( "HpIRInvFilterLength", iHpIRInvFilterLength, m_sfHpIRInv.length() );
95 
96  if( iHpIRInvFilterLength < m_sfHpIRInv.length() )
97  VA_WARN( "HPEQ reproduction module", "Filter length of given inverse HPIR file is longer than configured length" );
98 
99  // Calibrate ( HPTF * HPTFinv == 0dB )
100  m_sfHpIRInv.mul_scalar( float( dEqualizationCalibrationGain ) );
101 
102  double dSampleRate = m_oParams.pCore->GetCoreConfig()->oAudioDriverConfig.dSampleRate;
103  int iBlockLength = m_oParams.pCore->GetCoreConfig()->oAudioDriverConfig.iBuffersize;
104  m_pHPEQStreamFilter = new HPEQStreamFilter( dSampleRate, iBlockLength, m_sfHpIRInv );
105 }

284 bool HPEQStreamFilter::ExchangeFilter( const ITASampleFrame& sfHpIRinv )
285 {
286  if( sfHpIRinv.length() > pConvolverLeft->GetMaxFilterlength() )
287  return false;
288 
289  // Left
290  ITAUPFilter* pHPEQLeft = pConvolverLeft->RequestFilter();
291  pHPEQLeft->Load( (&sfHpIRinv[0])->data(), sfHpIRinv.length() );
292  pConvolverLeft->ExchangeFilter( pHPEQLeft );
293  pConvolverLeft->ReleaseFilter( pHPEQLeft );
294 
295  // Right
296  ITAUPFilter* pHPEQRight = pConvolverLeft->RequestFilter();
297  pHPEQRight->Load( (&sfHpIRinv[1])->data(), sfHpIRinv.length() );
298  pConvolverRight->ExchangeFilter( pHPEQRight );
299  pConvolverLeft->ReleaseFilter( pHPEQRight );
300 
301  return true;
302 }
