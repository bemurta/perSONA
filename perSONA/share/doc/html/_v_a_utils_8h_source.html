77  inline bool Translate( const std::string& s, T& dest ) const
78  {
79  std::string t = toUppercase( stripSpaces( s ) );
80  typename std::map< std::string, T >::const_iterator it = m_mLiterals.find( t );
81  if( it == m_mLiterals.end() )
82  return false;
83  dest = it->second;
84  return true;
