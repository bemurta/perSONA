66 int CVADirectivityManager::CreateDirectivity( const CVAStruct& oParams, const std::string& sName )
67 {
68  VA_VERBOSE( "DirectivityManager", "Creating directivity with name '" << sName << "'" );
69 
70  // Try to identify the directivity type by parsing parameter arguments
71  if( oParams.HasKey( "filepath" ) )
72  {
73  const std::string sFilePath = oParams[ "filepath" ];
74  const std::string sDestFilePath = m_pAssociatedCore->FindFilePath( sFilePath );
75  if( sDestFilePath.empty() )
76  VA_EXCEPT2( INVALID_PARAMETER, "Looked everywhere, but could not find file '" + sFilePath + "'" );
77 
78  // Format entscheiden und Implementierung wählen
79  IVADirectivity* pDirectivity = nullptr;
80  std::string sSuffix = toLowercase( getFilenameSuffix( sDestFilePath ) );
81 
82  if( sSuffix == "daff" )
83  {
84 
85  // Der DAFF-Reader wirft std::exceptions
86  DAFFReader* pReader = DAFFReader::create();
87 
88  int iError = pReader->openFile( sDestFilePath );
89  if( iError != DAFF_NO_ERROR )
90  {
91  delete pReader;
92  std::string sErrorStr = DAFFUtils::StrError( iError );
93  VA_EXCEPT1( std::string( "Could not load HRIR dataset from file \"" ) +
94  sDestFilePath + std::string( "\". " ) + sErrorStr + std::string( "." ) );
95  }
96 
97  if( pReader->getContentType() == DAFF_IMPULSE_RESPONSE )
98  {
99  delete pReader;
100  pDirectivity = new CVADirectivityDAFFHRIR( sDestFilePath, sName, m_dSampleRate );
101  }
102  else if( pReader->getContentType() == DAFF_MAGNITUDE_SPECTRUM )
103  {
104  delete pReader;
105  pDirectivity = new CVADirectivityDAFFEnergetic( sDestFilePath, sName );
106  }
107  else
108  {
109  delete pReader;
110  VA_EXCEPT2( INVALID_PARAMETER, "DAFF file does not contain recognized content type" );
111  }
112  }
113  else if( sSuffix == "xml" )
114  {
115  // Multipoles, etc
116  VA_EXCEPT_NOT_IMPLEMENTED;
117  }
118  else if( sSuffix == "sofa" )
119  {
120  // Sofa
121  VA_EXCEPT_NOT_IMPLEMENTED;
122  }
123  /*
124  else if (sSuffix == "hsh")
125  {
126  pHRIR = new CVAHRIRDatasetSH(sFilename, sName, m_dSamplerate);
127  }
128  */
129 
130  if( !pDirectivity )
131  VA_EXCEPT2( INVALID_PARAMETER, "Directiviy format not recognized or unsupported" );
132 
133  // HRIR hinzufügen
134  const int iID = m_oDirectivities.Add( pDirectivity );
135  VA_VERBOSE( "HRIRMan", "Successfully created directivity with ID " << iID );
136 
137  return iID;
138  }
139  else
140  {
141  VA_EXCEPT2( INVALID_PARAMETER, "Could not interpret parameters to create a directivity" );
142  }
143 }

145 bool CVADirectivityManager::DeleteDirectivity( const int iID )
146 {
147  VA_VERBOSE( "DirectivityManager", "Freeing HRIR with ID " << iID );
148 
149  // Prüfen ob die ID gültig ist
150  int iRefCount = m_oDirectivities.GetRefCount( iID );
151 
152  if( iRefCount != 0 )
153  {
154  // Noch Referenzen vorhanden oder ungültige ID (-1)
155  VA_VERBOSE( "DirectivityManager", "FreeHRIR failed, HRIR still in use by a user or invalid ID " << iID << ")" );
156  return false;
157  }
158 
159  // HRIR löschen
160  IVADirectivity* pHRIR = m_oDirectivities[ iID ];
161  delete pHRIR;
162  m_oDirectivities.Remove( iID );
163 
164  VA_VERBOSE( "DirectivityManager", "Directivity successfully cleared" );
165 
166  return true;
167 }
