63  const double tmp = ( 10.79586f * ( 1.0f - ( ( T_0 - 20.0f ) / T ) ) ) -
64  ( 5.02808f * log10( ( T / ( T_0 - 20.0f ) ) ) ) +
65  ( 1.50474f * 0.0001f * ( 1.0f - pow( ( double ) 10.0f, ( -8.29692f*( ( T / ( T_0 - 20.0f ) ) - 1.0f ) ) ) ) ) +

70  assert( 0.0f <= dHumidity && dHumidity <= 100.0f );
71  const double h = dHumidity * pow( ( double ) 10.0f, tmp ) / ( p_a / p_r );
72 
73  // Equation (3)
74  const double f_r_o = ( p_a / p_r ) * ( 24.0f + 4.04f*10.0e4*h*( 0.02f + h ) / ( 0.391f + h ) );
75 
76  // Equation (4)
77  const double f_r_n = ( p_a / p_r ) * pow( T / T_0, ( double ) -1.0f / 2.0f ) * ( 9.0f + 280.0f*h*exp( -4.710f * ( pow( T / T_0, ( double ) -( 1 / 3.0f ) ) - 1.0f ) ) );
78 
79  // Calculate for frequency bands
80  for( int i = 0; i < oMags.GetNumBands(); i++ )
81  {
82  // Frequency
83  double f = oMags.GetCenterFrequencies()[ i ];
84 
85  // Parts of Equation (5)
86  double dAlpha1 = 8.686f * pow( f, ( double ) 2.0f );
87  double dAlpha2 = 1.84f * 10e-11 * pow( p_a / p_r, ( double ) -1.0f ) * pow( T / T_0, ( double ) -1.0f / 2.0f );
88  double dAlpha3 = pow( T / T_0, ( double ) -5.0f / 2.0f );
89  double dAlpha4 = 0.01275f * exp( -2239.1f / T ) * pow( f_r_o + pow( f, ( double ) 2.0f ) / f_r_o, ( double ) -1.0f );
90  double dAlpha5 = 0.10680f * exp( -3352.0f / T ) * pow( f_r_n + pow( f, ( double ) 2.0f ) / f_r_n, ( double ) -1.0f );
91 
92  // Attenuation coefficient [dB/m], ~f, as assembly of Equation (5) parts
93  double dAlpha = dAlpha1 * ( dAlpha2 + dAlpha3 * ( dAlpha4 + dAlpha5 ) );
94 
95  // Resulting attenuation [dB], ~alpha (~f)
96  float fAttenuationAirAbsorptionDecibel = ( float ) dAlpha * dDistance;
97 
98  // Attenuation factor

100 
101  // Store decibel
102  oMags.SetMagnitude( i, fAttenuationAirAbsorptionDecibel );
103

105  }
106 
107  return;
