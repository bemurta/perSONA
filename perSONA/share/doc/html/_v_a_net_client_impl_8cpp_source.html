1670  IVistaTimerImp::SetSingleton( new VistaDefaultTimerImp );
1671 
1672  m_pProtocol = new CVANetNetworkProtocol;
1673 
1674  m_sServerIP = sServerAddress;
1675 
1676  VistaConnectionIP oRequestConnection( VistaConnectionIP::CT_TCP, m_sServerIP, iServerPort );
1677  if( oRequestConnection.GetIsConnected() == false )
1678  {
1679 #if VANET_CLIENT_VERBOSE==1
1680  std::cout << "VANetClient: Connection to server at [" << m_sServerIP << ":" << iServerPort << "] failed" << std::endl;
1681 #endif
1682  return VA_SERVER_CONNECTION_FAILED;
1683  }
1684  oRequestConnection.SetIsBlocking( true );
1685 
1686 
1687  // first, we read the first sent one to check if we have different endianess
1688  oRequestConnection.SetByteorderSwapFlag( VistaSerializingToolset::DOES_NOT_SWAP_MULTIBYTE_VALUES );
1689  VistaSerializingToolset::ByteOrderSwapBehavior bSwap = VistaSerializingToolset::DOES_NOT_SWAP_MULTIBYTE_VALUES;
1690  VANetCompat::sint32 iTest;
1691  oRequestConnection.ReadInt32( iTest );
1692  if( iTest != 1 )
1693  {
1694  bSwap = VistaSerializingToolset::SWAPS_MULTIBYTE_VALUES;
1695  oRequestConnection.SetByteorderSwapFlag( VistaSerializingToolset::SWAPS_MULTIBYTE_VALUES );
1696  }
1697 
1698  // Verify protocol version;
1699  VANetCompat::sint32 nMajor = -1;
1700  VANetCompat::sint32 nMinor = -1;
1701  oRequestConnection.ReadInt32( nMajor );
1702  oRequestConnection.ReadInt32( nMinor );
1703  if( nMajor != CVANetNetworkProtocol::VA_NET_PROTOCOL_VERSION_MAJOR || nMinor != CVANetNetworkProtocol::VA_NET_PROTOCOL_VERSION_MINOR )
1704  {
1705  oRequestConnection.WriteBool( false );
1706  oRequestConnection.WaitForSendFinish();
1707  Cleanup();
1708  return VA_PROTOCOL_INCOMPATIBLE;
1709  }
1710  else
1711  oRequestConnection.WriteBool( true );
1712 
1713  VANetCompat::sint32 iCommandChannelPort = 0;
1714  int iReturn = 0;
1715  iReturn = oRequestConnection.ReadInt32( iCommandChannelPort );
1716  if( iReturn != sizeof( VANetCompat::sint32 ) )
1717  {
1718  Cleanup();
1719  return VA_CONNECTION_ERROR;
1720  }
1721  if( iCommandChannelPort == -1 )
1722  {
1723  // server responded, but establishing the connection sockets failed
1724  Cleanup();
1725  return VA_CONNECTION_ERROR;
1726  }
1727  else if( iCommandChannelPort == -2 )
1728  {
1729  // server responded, but it doesn't accept any more connections
1730  Cleanup();
1731  return VA_SERVICE_IN_USE;
1732 }
1733  m_pCommandChannel = new VistaConnectionIP( VistaConnectionIP::CT_TCP, m_sServerIP, iCommandChannelPort );
1734 #ifdef VANET_CLIENT_SHOW_RAW_TRAFFIC
1735  m_pCommandChannel->SetShowRawSendAndReceive( true );
1736 #else
1737  m_pCommandChannel->SetShowRawSendAndReceive( false );
1738 #endif
1739 
1740  if( m_pCommandChannel->GetIsConnected() == false )
1741  {
1742 #ifdef VANET_CLIENT_VERBOSE
1743  std::cout << "VANetClient: Connecting command channel at [" << m_sServerIP << ":" << iCommandChannelPort << "] failed" << std::endl;
1744 #endif
1745  Cleanup();

1747  }
1748

1750  {

1752  {
1753  oRequestConnection.WriteInt32( 0 );
1754  m_pHeadChannel = NULL;
1755  break;
1756  }
1757  case VA_HC_SEPARATE_TCP:
1758  {
1759  oRequestConnection.WriteInt32( 1 );
1760  VANetCompat::sint32 iHeadPort = 0;
1761  oRequestConnection.ReadInt32( iHeadPort );

1763  m_sServerIP, iHeadPort );
1764  if( m_pHeadChannel->GetIsConnected() == false )
1765  {

1768  }
1769  break;

1777  m_sServerIP, iHeadPort );
1778  if( m_pHeadChannel->GetIsConnected() == false )
1779  {
1780  Cleanup();
1781  return VA_CONNECTION_ERROR;
1782  }

1788  return VA_UNKNOWN_ERROR;
1789  }
1790  }

1794  if( nRet != sizeof( VANetCompat::sint32 ) )

1797  Cleanup();

1803  Cleanup();
1804  return false;

1811  Cleanup();

1813  }

1816  m_pCommandChannel->SetIsBlocking( true );
1817  if( m_pHeadChannel )
1818  {
1819  m_pHeadChannel->SetByteorderSwapFlag( bSwap );
1820  m_pHeadChannel->SetIsBlocking( true );
1821  }
1822  m_pEventChannel->SetByteorderSwapFlag( m_pCommandChannel->GetByteorderSwapFlag() );
1823  m_pEventChannel->SetIsBlocking( true );
