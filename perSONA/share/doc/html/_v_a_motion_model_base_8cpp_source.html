78  m_oInputDataLog.log( oLogItem );
79  }
80 
81  // Sample Latest fuer Orientierung
82  m_vLastView = vView;
83  m_vLastUp = vUp;
84 
85  // Store new element
86  m_iTail = ( m_iTail + 1 ) % m_oConf.iNumHistoryKeys;
87  PositionVelocityKey& oNewKey = m_vKeys[ m_iTail ];
88  oNewKey.t = dTime;
89  oNewKey.p = vPos;
90  //oNewKey.v = vVel; // Get rid of velocity, this is not set by user anymore
91 
92  // Update number of elements
93  if( m_iNumKeys < m_oConf.iNumHistoryKeys )

99 bool CVABasicMotionModel::EstimatePosition( double dTime, VAVec3& vPos )
100 {
101  // Ensure monotonous growing query times - Maybe helpful with the model
102  if( m_iNumKeys == 0 )
103  return false;
104 
105  assert( dTime >= m_dLastQueryTime );
106  m_dLastQueryTime = dTime;
107 
108  MotionLogDataOutput oLogDataItem;
109  if( m_oConf.bLogEstimatedOutputEnabled )
110  {
111  oLogDataItem.dTime = dTime;
112  }
113 
114  int n = 0; // Number of considered history elements
115  double wsum = 0; // Sum of weights of considered history elements
116 
117  // Debug: iNumKeys = 1;
118  while( n < m_iNumKeys )
119  {
120  PositionVelocityKey* k = GetHistoryKey( n ); // Look n-times back into history
121  if( !k ) {
122  assert( false );
123  break;
124  }
125 
126  // Important: Estimation time must lie behind the latest key's time
127  double dt = ( double ) ( ( long double ) dTime - ( long double ) k->t );
128  //assert( dt > 0 );
129 
130  // Compute weight
131  k->w = ComputeWeight( dt );
132 
133  // Keyframe out of focus... We do not need to continue
134  if( k->w <= 0 ) break;
135 
136  // Perform linear extrapolation with this key
137  k->px.x = k->p.x + ( float ) dt*k->v.x;
138  k->px.y = k->p.y + ( float ) dt*k->v.y;
139  k->px.z = k->p.z + ( float ) dt*k->v.z;
140 
141  wsum += k->w;
142  n++;
143  }
144 
145  VAVec3 vVel;
146  vVel.Set( 0, 0, 0 );
147  vPos.Set( 0, 0, 0 );
148  // There are history keys, but all are out of the window range => Last value counts
149  if( ( n == 0 ) || ( wsum == 0 ) )
150  {
151  PositionVelocityKey* k = GetHistoryKey( 0 );
152  if( k )
153  {
154  double dt = ( double ) ( ( long double ) dTime - ( long double ) k->t );
155  vPos.x = k->p.x + ( float ) dt*k->v.x;
156  vPos.y = k->p.y + ( float ) dt*k->v.y;
157  vPos.z = k->p.z + ( float ) dt*k->v.z;
158  vVel.x = k->v.x;
159  vVel.y = k->v.y;
160  vVel.z = k->v.z;
161  }
162 
163  if( m_oConf.bLogEstimatedOutputEnabled )
164  {
165  oLogDataItem.vPos = vPos;
166  oLogDataItem.vView = m_vLastView; // hack: use last view and last up vector (no estimation)
167  oLogDataItem.vUp = m_vLastUp; // hack: use last view and last up vector (no estimation)
168  ConvertViewUpToQuaternion( m_vLastView, m_vLastUp, oLogDataItem.qOrient ); // hack: use last view and last up vector (no estimation)
169  oLogDataItem.iNumInvolvedKeys = 0;
170  m_oEstimationDataLog.log( oLogDataItem );
171  }
172  return true;
173  }
174 
175  // Superposition of all extrapolated positions according to their weight
176  for( int i = 0; i < n; i++ )
177  {
178  PositionVelocityKey* k = GetHistoryKey( i );
179  float s = ( float ) ( k->w / wsum );
180 
181  vPos.x += s * k->px.x;
182  vPos.y += s * k->px.y;
183  vPos.z += s * k->px.z;
184  vVel.x += k->v.x;
185  vVel.y += k->v.y;
186  vVel.z += k->v.z;
187  }
188 
189  if( m_oConf.bLogEstimatedOutputEnabled )
190  {
191  oLogDataItem.vPos = vPos;
192  oLogDataItem.iNumInvolvedKeys = n;
193  oLogDataItem.vView = m_vLastView; // hack: use last view and last up vector (no estimation)
194  oLogDataItem.vUp = m_vLastUp; // hack: use last view and last up vector (no estimation)
195  ConvertViewUpToQuaternion( m_vLastView, m_vLastUp, oLogDataItem.qOrient ); // hack: use last view and last up vector (no estimation)
196  m_oEstimationDataLog.log( oLogDataItem );
197  }
198  return true;
199 }

220 CVABasicMotionModel::PositionVelocityKey* CVABasicMotionModel::GetHistoryKey( int iLookback )
221 {
222  if( !( iLookback < m_iNumKeys ) ) {
223  assert( false );
224  return nullptr;
225  }
226 
227  assert( ( iLookback >= 0 ) && ( iLookback < m_oConf.iNumHistoryKeys ) );
228  int iIndex = ( m_iTail + m_oConf.iNumHistoryKeys - iLookback ) % m_oConf.iNumHistoryKeys;
229  assert( ( iIndex >= 0 ) && ( iIndex < ( int ) m_vKeys.size() ) );
230  return &m_vKeys[ iIndex ];

233 double CVABasicMotionModel::ComputeWeight( double t ) const
234 {
235  // Dreiecks-Funktion mit vorgegebener Fensterbreite
236  double w = 1 - abs( t - m_oConf.dWindowDelay ) / m_oConf.dWindowSize;
237  return ( w < 0 ? 0 : w );
