89 const float* CVAMachineSignalSource::GetStreamBlock( const CVAAudiostreamState* )
90 {
91  const int iNewTransition = m_iNewTransition;
92 
93  m_sbOut.Zero();
94 
95  // Missing sounds of machine not implemented yet
96  if( !( m_oConfig.bRequiresIdleSound && m_oConfig.bRequiresIdleSound && m_oConfig.bRequiresStopSound ) )
97  return m_sbOut.data();
98 
99  // Audiostream-synced data copy (todo use pointer)
100  if( m_bStartSoundNew )
101  {
102  m_sbStartSound = m_sbStartSoundNew;
103  m_bStartSoundNew = false;
104  }
105  if( m_bIdleSoundNew )
106  {
107  m_sbIdleSound = m_sbIdleSoundNew;
108  m_bIdleSoundNew = false;
109  }
110  if( m_bStopSoundNew )
111  {
112  m_sbStopSound = m_sbStopSoundNew;
113  m_bStopSoundNew = false;
114  }
115 
116  switch( m_iCurrentState )
117  {
118  case MACHINE_STATE_INVALID:
119  if( m_bHasStartSound || m_bHasIdleSound )
120  m_iCurrentState = MACHINE_STATE_STOPPED;
121  break;
122  case MACHINE_STATE_STOPPED:
123  if( iNewTransition == MACHINE_TRANSITION_START ) // Startup
124  {
125  int iSamplesWritten;
126  bool bEndReached;
127  AddSamplesStarting( m_sbOut, iSamplesWritten, bEndReached );
128 
129  if( bEndReached )
130  {
131  // Short starting sample, jump directly into idle state
132  AddSamplesIdling( m_sbOut, iSamplesWritten );
133  m_iCurrentState = MACHINE_STATE_IDLING;
134  }
135  else
136  {
137  m_iCurrentState = MACHINE_STATE_STARTING;
138  }
139  }
140  break;
141 
142  case MACHINE_STATE_STARTING:
143  if( iNewTransition == MACHINE_TRANSITION_HALT )
144  {
145  int iSamplesWritten;
146  bool bEndReached;
147  AddSamplesStopping( m_sbOut, iSamplesWritten, bEndReached );
148 
149  if( bEndReached )
150  {
151  m_iCurrentState = MACHINE_STATE_STOPPED;
152  ResetReadCursors();
153  }
154  }
155  else if( iNewTransition == MACHINE_TRANSITION_START )
156  {
157  int iSamplesWritten;
158  bool bEndReached;
159  AddSamplesStarting( m_sbOut, iSamplesWritten, bEndReached );
160 
161  if( bEndReached )
162  {
163  m_iCurrentState = MACHINE_STATE_IDLING;
164  AddSamplesIdling( m_sbOut, iSamplesWritten );
165  }
166  }
167  else
168  {
169  assert( false );
170  }
171  break;
172 
173  case MACHINE_STATE_IDLING:
174  if( iNewTransition == MACHINE_TRANSITION_HALT )
175  {
176  AddSamplesIdling( m_sbOut, false );
177 
178  int iSamplesWritten;
179  bool bEndReached;
180  ITASampleBuffer sbOutTemp( m_sbOut );
181  AddSamplesStopping( sbOutTemp, iSamplesWritten, bEndReached );
182 
183  if( m_bCrossfadeSounds )
184  {
185  // Fade in stopping sample buffer
186  m_sbOut.Crossfade( sbOutTemp, 0, iSamplesWritten, ITA_FADE_IN, ITASampleBuffer::COSINE_SQUARE );
187  }
188  else
189  {
190  m_sbOut.write( sbOutTemp, iSamplesWritten );
191  if( iSamplesWritten < m_sbOut.length() )
192  m_sbOut.Zero( iSamplesWritten, m_sbOut.length() - iSamplesWritten );
193  }
194 
195  if( bEndReached )
196  {
197  m_iCurrentState = MACHINE_STATE_STOPPED;
198  ResetReadCursors();
199  }
200  else
201  {
202  m_iCurrentState = MACHINE_STATE_HALTING;
203  }
204  }
205  else
206  {
207  // Idling ...
208  AddSamplesIdling( m_sbOut, false );
209  }
210  break;
211 
212  case MACHINE_STATE_HALTING:
213  int iSamplesWritten;
214  bool bEndReached;
215  AddSamplesStopping( m_sbOut, iSamplesWritten, bEndReached );
216  if( bEndReached )
217  {
218  m_iCurrentState = MACHINE_STATE_STOPPED;
219  ResetReadCursors();
220  }
221  break;
222 
223  default:
224  m_sbOut.Zero();
225 
226  }
227 
228  for( int i=0; i<m_sbOut.length(); i++ )
229  {
230  if( m_sbOut[i] > 1.0f || m_sbOut[i] < -1.0f )
231  VA_WARN( "MachineSignalSource", "high energy detected" );
232  }
233 
234  return m_sbOut.data();
235 }

483 void CVAMachineSignalSource::AddSamplesIdling( ITASampleBuffer& sbOut, int iOutputOffset/*=0 */ )
484 {
485  if( !m_bHasIdleSound )
486  {
487  return;
488  }
489 
490  assert( sbOut.length() == int( GetBlocklength() ) );
491  assert( iOutputOffset >= 0 && iOutputOffset < sbOut.length() );
492  assert( iOutputOffset < sbOut.length() );
493 
494  int iSize;
495  if( m_oConfig.bDisableInterpolation == true )
496  {
497  iSize = GetBlocklength() - iOutputOffset;
498  m_sbIdleSound.cyclic_read( sbOut.data(), iSize, m_iIdlingSoundCursor );
499  }
500  else
501  {
502  int iLeft, iRight;
503  m_pInterpRoutine->GetOverlapSamples( iLeft, iRight );
504  iSize = int( m_fMachineSpeed * GetBlocklength() ) - iOutputOffset;
505  assert( iLeft + iSize + iRight <= m_sbInterpolationSrc.length() );
506  int iSrcOffset = m_iIdlingSoundCursor - iLeft;
507  if( iSrcOffset < 0 )
508  iSrcOffset += m_sbIdleSound.length();
509  m_sbIdleSound.cyclic_read( m_sbInterpolationSrc.data(), iLeft + iSize + iRight, iSrcOffset ); // copy
510 
511  int iInputLength = iLeft+iSize;
512  m_pInterpRoutine->Interpolate( &m_sbInterpolationSrc, iInputLength, iLeft, &sbOut, sbOut.length() - iOutputOffset, iOutputOffset );
513  }
514 
515  m_iIdlingSoundCursor += iSize;
516  if( m_iIdlingSoundCursor >= m_sbIdleSound.length() )
517  m_iIdlingSoundCursor = m_iIdlingSoundCursor % m_sbIdleSound.length();
518 }
