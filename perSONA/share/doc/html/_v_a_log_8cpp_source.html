113 bool CVARealtimeLogger::LoopBody()
114 {
115 
116  if( ( m_iTriggerCnt == 0 ) && !m_bStop ) {
117  // Blockierend warten
118  m_evTrigger.WaitForEvent( true );
119 
120  // Daten greifen
121  std::vector<CVALogItem> vLogItems;
122  m_mxRegistration.Lock();
123  std::list<CVARealtimeLogStream*>::const_iterator it;
124  for( it = m_lpStreams.begin(); it != m_lpStreams.end(); ++it )
125  {
126  CVARealtimeLogStream* pStream = *it;
127  CVALogItem oLogItem;
128  while( pStream->m_qLog.try_pop( oLogItem ) )
129  vLogItems.push_back( oLogItem );
130  }
131  m_mxRegistration.Unlock();
132 
133  // Daten sortieren (mittels überlademen Vergleichsoperator
134  if( vLogItems.size() > 1 )
135  std::sort( vLogItems.begin(), vLogItems.end() );
136 
137  // Daten ausgeben
138 
139  // Trigger-Event zurücksetzen
140  m_evTrigger.ResetThisEvent();
141  }
142 
143  // Stop-Flag überprüfen, ggf. selbst beenden
144  if( m_bStop ) {
145  IndicateLoopEnd();
146  return false;
147  }
148 
149  m_iTriggerCnt = 0;
150 
151  // Kein Wechsel zu einem Thread (no yield)
152  return false;
153 }
