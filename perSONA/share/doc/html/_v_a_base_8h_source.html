100  inline VAVec3 Cross( const VAVec3& vCrossProduct ) const
101  {
102  VAVec3 vCrossResult;
103  vCrossResult.x = y * vCrossProduct.z - z * vCrossProduct.y;
104  vCrossResult.y = z * vCrossProduct.x - x * vCrossProduct.z;
105  vCrossResult.z = x * vCrossProduct.y - y * vCrossProduct.x;
106  return vCrossResult;
107  };
