136 int IVAInterface::ParseAuralizationModeStr( const std::string& sModeStr, const int iBaseMode )
137 {
138  // Remove all whitespaces, valid chars: Alphabetic + *+-,
139  std::string s;
140  for( size_t k = 0; k < sModeStr.size(); k++ )
141  {
142  if( isspace( sModeStr[ k ] ) )
143  continue;
144 
145  if( isalpha( sModeStr[ k ] ) )
146  {
147  s += char( toupper( sModeStr[ k ] ) );
148  continue;
149  }
150 
151  if( ( sModeStr[ k ] == '*' ) ||
152  ( sModeStr[ k ] == '+' ) ||
153  ( sModeStr[ k ] == '-' ) ||
154  ( sModeStr[ k ] == ',' ) )
155  {
156  s += char( toupper( sModeStr[ k ] ) );
157  continue;
158  }
159 
160  VA_EXCEPT2( INVALID_PARAMETER, "Auralization mode specification contains invalid characters" );
161  }
162 
163  // Trivial cases: Empty string, "null", "none", "default", "all"
164  if( s.empty() ) return VA_AURAMODE_NOTHING;
165  if( ( s == "NULL" ) || ( s == "NONE" ) ) return VA_AURAMODE_NOTHING;
166  if( s == "DEFAULT" ) return VA_AURAMODE_DEFAULT;
167  if( ( s == "ALL" ) || ( s == "*" ) ) return VA_AURAMODE_ALL;
168 
169  // Format: List of tokens seperated by commas (possibly whitespaces)
170  // [fwe] For not adding PCRE this is implemented by hand here
171  size_t i = 0, j = 0, op = 0;
172  bool err = false;
173  int m = 0; // Scanning modes: 0 => none, 1 => token
174  int def = 0, plus = 0, minus = 0;
175  std::string sErrorMsg;
176 
177  for( i; i < s.length(); i++ )
178  {
179  if( isalpha( s[ i ] ) )
180  {
181  if( m == 0 )
182  {
183  // None => Begin token
184  j = i;
185  m = 1;
186  }
187  else
188  {
189  // Token => extend
190  continue;
191  }
192  }
193 
194  if( s[ i ] == '+' )
195  {
196  if( ( m == 1 ) || ( op != 0 ) )
197  { // Scanning toking || multiple operands => Error
198  err = true;
199  sErrorMsg = "Multiple plus operand found in auralisation mode";
200  break;
201  }
202 
203  op = 1;
204  }
205 
206  if( s[ i ] == '-' )
207  {
208  if( ( m == 1 ) || ( op != 0 ) )
209  { // Scanning toking || multiple operands => Error
210  err = true;
211  sErrorMsg = "Multiple minus operand found in auralisation mode";
212  break;
213  }
214 
215  op = 2;
216  }
217 
218  if( s[ i ] == ',' )
219  {
220  if( m == 0 )
221  { // No token => Error
222  err = true;
223  sErrorMsg = "No token in auralisation mode found";
224  break;
225  }
226  else
227  { // Finished token
228  std::string t = s.substr( j, i - j );
229  int p = ParseAuralizationModeToken( t );
230  if( p == -1 ) {
231  // Invalid token
232  sErrorMsg = "Invalid token '" + t + "' found in auralisation mode";
233  err = true;
234  break;
235  }
236 
237  if( op == 0 ) def |= p; // No operator
238  if( op == 1 ) plus |= p; // Plus operator
239  if( op == 2 ) minus |= p; // Minus operator
240 
241  op = 0;
242  j = 0;
243  m = 0;
244  }
245  }
246  }
247 
248  // Unfinished token?
249  if( m != 0 )
250  {
251  std::string t = s.substr( j, i - j );
252  int p = ParseAuralizationModeToken( t );
253  if( p == -1 )
254  {
255  // Invalid token
256  sErrorMsg = "Invalid token '" + t + "' found in auralisation mode";
257  err = true;
258  }
259 
260  if( op == 0 ) def |= p; // No operator
261  if( op == 1 ) plus |= p; // Plus operator
262  if( op == 2 ) minus |= p; // Minus operator
263  }
264 
265  if( err )
266  VA_EXCEPT2( INVALID_PARAMETER, "Invalid auralization mode specification, " + sErrorMsg );
267 
268 
269  // Combine modes:
270  int result = ( iBaseMode & VA_AURAMODE_ALL );
271  if( def != 0 ) // Assignment => Forget about the base
272  result = def;
273 
274  // Plus flags
275  result |= plus;
276 
277  // Minus flags (stronger than plus)
278  result &= ~minus;
279 
280  return result;
281 }

284 {
285  std::stringstream ss;
286 
287  if( !GetAuralizationModeValid( iAuralizationMode ) )

292 
293  if( bShortFormat )

296

298  if( iAuralizationMode & VA_AURAMODE_EARLY_REFLECTIONS ) ss << "ER, ";
299  if( iAuralizationMode & VA_AURAMODE_DIFFUSE_DECAY ) ss << "DD, ";
300  if( iAuralizationMode & VA_AURAMODE_SOURCE_DIRECTIVITY ) ss << "DIR, ";
301  if( iAuralizationMode & VA_AURAMODE_MEDIUM_ABSORPTION ) ss << "AA, ";
302  if( iAuralizationMode & VA_AURAMODE_TEMP_VAR ) ss << "TV, ";
303  if( iAuralizationMode & VA_AURAMODE_SCATTERING ) ss << "SC, ";
304  if( iAuralizationMode & VA_AURAMODE_DIFFRACTION ) ss << "DIF, ";
305  if( iAuralizationMode & VA_AURAMODE_NEARFIELD ) ss << "NF, ";
306  if( iAuralizationMode & VA_AURAMODE_DOPPLER ) ss << "DP, ";
307  if( iAuralizationMode & VA_AURAMODE_SPREADING_LOSS ) ss << "SL, ";
308  if( iAuralizationMode & VA_AURAMODE_TRANSMISSION ) ss << "TR, ";
309  if( iAuralizationMode & VA_AURAMODE_ABSORPTION ) ss << "AB, ";
310  }
311  else
312  {
313  if( iAuralizationMode == 0 ) return "None (0)";
314 
315  if( iAuralizationMode & VA_AURAMODE_DIRECT_SOUND ) ss << "direct sound, ";
316  if( iAuralizationMode & VA_AURAMODE_EARLY_REFLECTIONS ) ss << "early reflections, ";
317  if( iAuralizationMode & VA_AURAMODE_DIFFUSE_DECAY ) ss << "diffuse decay, ";
318  if( iAuralizationMode & VA_AURAMODE_SOURCE_DIRECTIVITY ) ss << "source directivity, ";
319  if( iAuralizationMode & VA_AURAMODE_MEDIUM_ABSORPTION ) ss << "air absorption, ";
320  if( iAuralizationMode & VA_AURAMODE_TEMP_VAR ) ss << "atmospheric temporal variations, ";
321  if( iAuralizationMode & VA_AURAMODE_SCATTERING ) ss << "scattering, ";
322  if( iAuralizationMode & VA_AURAMODE_DIFFRACTION ) ss << "diffraction, ";
323  if( iAuralizationMode & VA_AURAMODE_NEARFIELD ) ss << "near-field effects, ";
324  if( iAuralizationMode & VA_AURAMODE_DOPPLER ) ss << "doppler shifts, ";
325  if( iAuralizationMode & VA_AURAMODE_SPREADING_LOSS ) ss << "spherical spreading loss, ";
326  if( iAuralizationMode & VA_AURAMODE_TRANSMISSION ) ss << "transmission, ";
327  if( iAuralizationMode & VA_AURAMODE_ABSORPTION ) ss << "absorption, ";
328  }
329 
330  // Remove last ,_
331  std::string s( ss.str() );
332  s[ 0 ] = char( toupper( s[ 0 ] ) );
333  return s.substr( 0, s.length() - 2 );
334 }
