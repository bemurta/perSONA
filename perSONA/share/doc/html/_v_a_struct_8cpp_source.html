70 CVAStructValue* CVAStruct::Traverse( const std::string& sPath, size_t iPathCursor, const char cPathSeparator )
71 {
72  // Find next path element
73  size_t n;
74  n = sPath.find( cPathSeparator, iPathCursor );
75  if( n == std::string::npos )
76  {
77  // Last element on path
78  Map::iterator it = m_mData.find( sPath.substr( iPathCursor, sPath.length() - iPathCursor ) );
79  return ( it != m_mData.end() ? &( it->second ) : nullptr );
80  }
81  else
82  {
83  // Next element on path (at least one more follows)
84  std::string t = sPath.substr( iPathCursor, n - iPathCursor );
85  Map::iterator it = m_mData.find( t );
86  if( it == m_mData.end() )
87  return nullptr; // Error: Path not found
88 
89  CVAStructValue& oNode = it->second;
90  if( !oNode.IsStruct() )
91  return nullptr; // Error: Node is a leaf, paths goes further...
92 
93  // Traverse into subtree
94  return oNode.GetStruct().Traverse( sPath, n + 1, cPathSeparator );
95  }
96 }

220 std::string CVAStruct::ToString( const int iIndent ) const
221 {
222  std::stringstream ss;
223  std::string margin( iIndent, ' ' );
224 
225  for( Map::const_iterator cit = m_mData.begin(); cit != m_mData.end(); ++cit )
226  {
227  const CVAStructValue& k = cit->second;
228  switch( k.iDatatype )
229  {
230  case CVAStructValue::BOOL:
231  {
232  ss << margin << cit->first << " = " << cit->second.ToString() << " [bool]" << std::endl;
233  break;
234  }
235  case CVAStructValue::INT:
236  {
237  ss << margin << cit->first << " = " << cit->second.ToString() << " [int]" << std::endl;
238  break;
239  }
240  case CVAStructValue::DOUBLE:
241  {
242  ss << margin << cit->first << " = " << cit->second.ToString() << " [double]" << std::endl;
243  break;
244  }
245  case CVAStructValue::STRING:
246  {
247  ss << margin << cit->first << " = \"" << cit->second.ToString() << "\" [string]" << std::endl;
248  break;
249  }
250  case CVAStructValue::STRUCT:
251  {
252  const CVAStruct& oKey = cit->second;
253  ss << margin << cit->first << " = [struct] { " << std::endl
254  << margin << oKey.ToString( iIndent + 4 )
255  << margin << "}" << std::endl;
256  break;
257  }
258  case CVAStructValue::DATA:
259  {
260  ss << margin << cit->first << " = [data: " << cit->second.GetDataSize() << " bytes]" << std::endl;
261  break;
262  }
263  case CVAStructValue::SAMPLEBUFFER:
264  {
265  ss << margin << cit->first << " = [samplebuffer: " << cit->second.ToString() << " samples]" << std::endl;
266  break;
267  }
268  }
269  }
270 
271  return ss.str();
272 }

429 CVAStructValue& CVAStructValue::operator=( const CVAStructValue& rhs )
430 {
431  // Take care of self-assignment!
432  if( this == &rhs )
433  return *this;
434 
435  // Free associated struct first
436  delete pStruct;
437  iDataSize = 0;
438 
439  // Fast assignment. Copy only what is necessary ...
440  iDatatype = rhs.iDatatype;
441  switch( iDatatype )
442  {
443  case BOOL:
444  bValue = rhs.bValue;
445  break;
446 
447  case INT:
448  iValue = rhs.iValue;
449  break;
450 
451  case DOUBLE:
452  dValue = rhs.dValue;
453  break;
454 
455  case STRING:
456  sValue = rhs.sValue;
457  break;
458 
459  case STRUCT:
460  // Autonome Kopie der Schüssel Struktur erzeugen
461  pStruct = new CVAStruct( *rhs.pStruct );
462  break;
463 
464  case DATA:
465  // Autonome Kopie des Schüssel BLOBs erzeugen
466  SetData( rhs.GetData(), rhs.iDataSize );
467  break;
468 
469  case SAMPLEBUFFER:
470  sbValue = rhs.sbValue;
471  break;
472  }
473 
474  return *this;
475 }

533 CVAStructValue::operator bool() const
534 {
535  switch( iDatatype )
536  {
537  case BOOL:
538  // bool = Key[bool] obligatory.
539  return bValue;
540 
541  case INT:
542  // bool = Key[int] is allowed. Default semantic like in C/C++ (false == 0).
543  return ( iValue != 0 );
544 
545  case DOUBLE:
546  // bool = Key[double] is imprecise and but not forbidden. Same is for int.
547  return ( dValue != 0 );
548 
549  case STRING:
550  // bool = Key[string] may be possible.
551  {
552  std::string s( sValue );
553  std::transform( s.begin(), s.end(), s.begin(), ( int( *)( int ) ) ::toupper );
554  if( ( s == "no" ) || ( s == "false" ) )
555  return false;
556  if( ( s == "yes" ) || ( s == "true" ) )
557  return true;
558  }
559 
560  VA_EXCEPT1( std::string( "Key \"" ) + ( *psKey ) + std::string( "\" cannot be interpreted as an logical value" ) );
561 
562  case SAMPLEBUFFER:
563  return ( sbValue.GetNumSamples() > 0 );
564  case STRUCT:
565  // bool = Key[struct] is not possible.
566  VA_EXCEPT1( "Types cannot be converted" );
567 
568  case UNASSIGNED:
569  // Key is unassigned
570  VA_EXCEPT1( std::string( "Key \"" ) + ( *psKey ) + std::string( "\" does not exist" ) );
571  }
572 
573  // Invalid datatype.
574  assert( false );
575  return false;
576 }
