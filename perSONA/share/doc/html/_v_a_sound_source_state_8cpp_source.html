78 CVAMotionState* CVASoundSourceState::AlterMotionState()
79 {
80  assert( !IsFixed() );
81 
82  // Falls der Zustand finalisiert => Zustand aus der Basiskonfig => Autonomen Zustand erzeugen
83  // Falls der Zustand nicht finalisiert => Bereits erzeugter autonomen Zustand => Diesen zurückgeben
84 
85  if( !data.pMotionState )
86  {
87  data.pMotionState = GetManager()->RequestMotionState();
88  data.pMotionState->Initialize( GetModificationTime() );
89  }
90  else
91  {
92  if( data.pMotionState->IsFixed() )
93  {
94  // Autonomen Zustand ableiten
95  CVAMotionState* pNewState = GetManager()->RequestMotionState();
96 
97  // Änderungszeit des übergeordneten Szenezustands übernehmen
98  pNewState->Copy( data.pMotionState, GetModificationTime() );
99  data.pMotionState->RemoveReference();
100 
101  data.pMotionState = pNewState;
102  }
103  }
104 
105  return data.pMotionState;
106 }
