94  {

97  if( m_bGBPFirst )

100

102 
103  if( m_oState.i64Sample == 0 )
104  {
105  // Start
106  m_dStreamStartTime = m_pClock->getTime();
107  m_fCurrentClockOffset = fNewClockOffset;
108  m_oState.bTimeReset = false;
109  }
110  else
111  {
112  m_oState.bTimeReset = ( m_fCurrentClockOffset != fNewClockOffset );
113  m_fCurrentClockOffset = fNewClockOffset;
114  }
115 
116  // Clock drift compensation factor for RME Hammerfall - via manual measurement @todo remove
117  m_oState.dSysTime = m_dStreamStartTime + ( double ) ( m_oState.i64Sample + m_uiBlocklength ) / m_dSamplerate * 0.999978;
118  m_oState.dCoreTime = m_oState.dSysTime - ( double ) m_fCurrentClockOffset;
119  m_oState.bSyncMod = ( ( *m_plSyncModOwner ) != -1 );
120 
121  m_bGBPFirst = false;
122 
123  m_pSignalSourceMan->FetchInputData( &m_oState );
124  }
125 
126  return m_pSource->GetBlockPointer( uiChannel, &m_oState );
127  }
