394

397  double dcos = cos( k / iBlocklength*( ITAConstants::PI_F / 2 ) );
398  pfOutputData[ k ] += pfInputData[ k ] * float( ( dcos * dcos * ( vdLoudspeakerLastGains[ j ] - vdLoudspeakerGains[ j ] ) + vdLoudspeakerGains[ j ] ) );
399  }
400

427 void CVAVBAPFreeFieldAudioRenderer::CalculateInverseMatrix3x3( const double *dOriginalMatrix, double *dInverseMatrix ) const
428 {
429  // Matritzen werden zeilenweise gelesen/geschrieben
430 
431  // bennene Koeffizienten
432  //Zeile 1
433  const double& a11 = dOriginalMatrix[ 0 ];
434  const double& a12 = dOriginalMatrix[ 1 ];
435  const double& a13 = dOriginalMatrix[ 2 ];
436  //Zeile 2
437  const double& a21 = dOriginalMatrix[ 3 ];
438  const double& a22 = dOriginalMatrix[ 4 ];
439  const double& a23 = dOriginalMatrix[ 5 ];
440  //Zeile 3
441  const double& a31 = dOriginalMatrix[ 6 ];
442  const double& a32 = dOriginalMatrix[ 7 ];
443  const double& a33 = dOriginalMatrix[ 8 ];
444 
445  // finde Determinante
446  double dDetOriginalMatrix = ( a11*a22*a33 ) + ( a12*a23*a31 ) + ( a13*a21*a32 ) - ( a13*a22*a31 ) - ( a12*a21*a33 ) - ( a11*a23*a32 );
447 
448  //Koeffizienten der Transponierten
449  //Zeile 1
450  const double& t11 = a11;
451  const double& t12 = a21;
452  const double& t13 = a31;
453  //Zeile 2
454  const double& t21 = a12;
455  const double& t22 = a22;
456  const double& t23 = a32;
457  //Zeile 3
458  const double& t31 = a13;
459  const double& t32 = a23;
460  const double& t33 = a33;
461 
462 
463  //Finde die Determinanten der Untermatrix
464  //Zeile 1
465  dInverseMatrix[ 0 ] = t22*t33 - t23*t32;
466  dInverseMatrix[ 1 ] = t21*t33 - t23*t31;
467  dInverseMatrix[ 2 ] = t21*t32 - t22*t31;
468  //Zeile 2
469  dInverseMatrix[ 3 ] = t12*t33 - t13*t32;
470  dInverseMatrix[ 4 ] = t11*t33 - t13*t31;
471  dInverseMatrix[ 5 ] = t11*t32 - t12*t31;
472  //Zeile 3
473  dInverseMatrix[ 6 ] = t12*t23 - t13*t22;
474  dInverseMatrix[ 7 ] = t11*t23 - t13*t21;
475  dInverseMatrix[ 8 ] = t11*t22 - t12*t21;
476 
477 
478  // Bilde neue Matrix mit Vorzeichen
479  //Zeile 1
480  dInverseMatrix[ 0 ] /= dDetOriginalMatrix;
481  dInverseMatrix[ 1 ] /= -dDetOriginalMatrix;
482  dInverseMatrix[ 2 ] /= dDetOriginalMatrix;
483  //Zeile 2
484  dInverseMatrix[ 3 ] /= -dDetOriginalMatrix;
485  dInverseMatrix[ 4 ] /= dDetOriginalMatrix;
486  dInverseMatrix[ 5 ] /= -dDetOriginalMatrix;
487  //Zeile 3
488  dInverseMatrix[ 6 ] /= dDetOriginalMatrix;
489  dInverseMatrix[ 7 ] /= -dDetOriginalMatrix;
490  dInverseMatrix[ 8 ] /= dDetOriginalMatrix;
491 }

600 bool CVAVBAPFreeFieldAudioRenderer::CalculateLoudspeakerGains2D( const VAVec3& vSoundSource, std::vector< double >& vdLoudspeakerGains ) const
601 {
602  std::vector< int > viMinIDs; //Die Indices der zu nutzenden Lautsprecher
603  viMinIDs.resize( 2 );
604  std::vector <double> vdMinValues; //Der Abstand des Lautsprecher zur Quelle
605  vdMinValues.resize( 2 );
606 
607  for( int j = 0; j < 2; j++ )
608  {
609  viMinIDs[ j ] = -1;
610  vdMinValues[ j ] = 8e9; //unendlich wert spezifizieren (optional)
611  }
612 
613  // Aktive Lautsprecher finden 2D
614  for( size_t j = 0; j < m_voLoudspeaker.size(); j++ ) // Schleife über LS
615  {
616  // Entfernung berechnen
617  VistaVector3D vDifference = m_voLoudspeaker[ j ].pos - VistaVector3D( vSoundSource.comp );
618  double dDistance = vDifference.GetLength();
619 
620  if( dDistance < *max_element( vdMinValues.begin(), vdMinValues.end() ) )
621  {
622  // @todo: Warning klären ...
623  int iIDChange = int( max_element( vdMinValues.begin(), vdMinValues.end() ) - vdMinValues.begin() );
624  viMinIDs[ iIDChange ] = int( j );
625  vdMinValues[ iIDChange ] = dDistance;
626  }
627  }
628 
629  int iLS1 = viMinIDs[ 0 ];
630  int iLS2 = viMinIDs[ 1 ];
631 
632  const VistaVector3D &posLS1 = m_voLoudspeaker[ iLS1 ].pos;
633  const VistaVector3D &posLS2 = m_voLoudspeaker[ iLS2 ].pos;
634 
635  if( ( viMinIDs[ 0 ] == -1 ) || ( viMinIDs[ 1 ] == -1 ) )
636  return false;
637 
638  // Berechnung der Gains über die Inverse
639  double dDetA = ( -posLS1[ 2 ] * posLS2[ 0 ] ) - ( -posLS2[ 2 ] * posLS1[ 0 ] );
640 
641  if( dDetA == 0.0f )
642  return false;
643 
644  // Berechnung der Gains nach Pulkki
645  vdLoudspeakerGains[ viMinIDs[ 0 ] ] = ( 1 / dDetA*( ( -vSoundSource.z*posLS2[ 0 ] ) + ( vSoundSource.x*posLS2[ 2 ] ) ) );
646  vdLoudspeakerGains[ viMinIDs[ 1 ] ] = ( 1 / dDetA*( ( -vSoundSource.z*-posLS1[ 0 ] ) + ( vSoundSource.x*-posLS1[ 2 ] ) ) );
647 
648  // Normalisierung nach Pulkki
649  double dCorrection = sqrt( vdLoudspeakerGains[ viMinIDs[ 0 ] ] * vdLoudspeakerGains[ viMinIDs[ 0 ] ] + vdLoudspeakerGains[ viMinIDs[ 1 ] ] * vdLoudspeakerGains[ viMinIDs[ 1 ] ] );
650  vdLoudspeakerGains[ viMinIDs[ 0 ] ] /= dCorrection;
651  vdLoudspeakerGains[ viMinIDs[ 1 ] ] /= dCorrection;
652 
653  return true;
654 }
