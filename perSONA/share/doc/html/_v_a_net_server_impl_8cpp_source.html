144  inline bool AddObservedConnection( VistaConnectionIP* pConnection )
145  {
146  m_mapConnections[ m_iIDCounter ] = pConnection;
147  if( m_oConnMultiplexer.AddMultiplexPoint( pConnection->GetConnectionWaitForDescriptor(),
148  m_iIDCounter, VistaIOMultiplexer::eIODir(
149  VistaIOMultiplexer::MP_IOALL ) ) == false )
150  {
151  m_mapConnections.erase( m_mapConnections.find( m_iIDCounter ) );

158  inline bool RemoveObservedConnection( VistaConnectionIP* pConnection )
159  {
160  std::map<int, VistaConnectionIP*>::iterator itEntry = m_mapConnections.begin();
161  bool bFound = false;
162  while( itEntry != m_mapConnections.end() )
163  {
164  if( ( *itEntry ).second == pConnection )
165  {
166  bFound = true;
167  break;
168  }
169  else
170  {
171  ++itEntry;
172  }
173  }
174 
175  if( bFound == false )
176  return false;
177 
178  if( m_oConnMultiplexer.RemMultiplexPointByTicket( ( *itEntry ).first ) == false )
179  return false;
180 
181  m_mapConnections.erase( itEntry );
182  return true;
183  };
184 
185  inline bool LoopBody()
186  {
187  int nUpdateID = m_oConnMultiplexer.Demultiplex();
188  if( nUpdateID == 0 )
189  return false;
190  else if( nUpdateID < 0 )
191  {
192  std::cerr << "VA Server: Connection Observer encountered Demultiplex error" << std::endl;
193  IndicateLoopEnd();
194  return false;
195  }
196 
197  try
198  {
199  m_mapConnections[ nUpdateID ]->SetWaitForDescriptorEventSelectIsEnabled( false );
200  if( m_pProtocol->ProcessMessageFromClient( m_mapConnections[ nUpdateID ] ) == false )
201  {
202  m_pParent->RemoveClient( m_mapConnections[ nUpdateID ], CVANetNetworkProtocol::VA_NET_SERVER_DISCONNECT );
203  }
204  else
205  {
206  if( m_mapConnections[ nUpdateID ] )
207  m_mapConnections[ nUpdateID ]->SetWaitForDescriptorEventSelectIsEnabled( true );
208  }
209  }
210 #ifdef VANET_SERVER_VERBOSE
211  catch( CVAException& e )
212  {
213  std::cerr << "VA Server: caught exception and will disconnect now (" << e << ")" << std::endl;
214  m_pParent->RemoveClient( m_mapConnections[ nUpdateID ], CVANetNetworkProtocol::VA_NET_SERVER_DISCONNECT );
215  }
216 #else
217  catch( CVAException& )
218  {
219  m_pParent->RemoveClient( m_mapConnections[ nUpdateID ], CVANetNetworkProtocol::VA_NET_SERVER_DISCONNECT );
220  }
221 #endif
222  return false;
223  };

539 int CVANetServerImpl::AddNewClient( VistaConnectionIP* pTemporaryConnection )
540 {
541  VistaMutexLock oLock( *m_pServerChangeLock );
542 
543  pTemporaryConnection->WriteInt32( 1 ); // we write an initial defined value
544  // to check if there's an endianess difference
545 
546  // we write our protocol to ensure that it is the same
547  pTemporaryConnection->WriteInt32( ( VANetCompat::sint32 )CVANetNetworkProtocol::VA_NET_PROTOCOL_VERSION_MAJOR );
548  pTemporaryConnection->WriteInt32( ( VANetCompat::sint32 )CVANetNetworkProtocol::VA_NET_PROTOCOL_VERSION_MINOR );
549 
550  bool bVersionMatch = false;
551  pTemporaryConnection->ReadBool( bVersionMatch );
552  if( bVersionMatch == false )
553  return VA_CONNECTION_FAILED;
554 
555  CClientData oNewData;
556 
557  if( m_iMaxNumClients >= 0 && ( int ) m_vecClients.size() >= m_iMaxNumClients )
558  {
559  std::cerr << "VA Server: Client applied, but too many clients already connected" << std::endl;
560  pTemporaryConnection->WriteInt32( -2 );
561  return VA_CONNECTION_FAILED;
562  }
563 
564  VANetCompat::sint32 iCommandPort = GetNextFreePort();
565  if( iCommandPort == -1 )
566  {
567  std::cerr << "VA Server: Client applied, but no more free ports available" << std::endl;
568  pTemporaryConnection->WriteInt32( -1 );
569  return VA_CONNECTION_FAILED;
570  }
571 
572  VistaTCPServer oCommandServer( m_sServerIP, iCommandPort );
573  if( oCommandServer.GetIsValid() == false )
574  {
575  std::cerr << "VA Server: Client applied, but could not open "
576  << "command socket on port [" << iCommandPort << "]" << std::endl;
577  pTemporaryConnection->WriteInt32( -1 );
578  return VA_CONNECTION_FAILED;
579  }
580 
581  pTemporaryConnection->WriteInt32( iCommandPort );
582 
583  VistaTCPSocket* pCommandSocket = oCommandServer.GetNextClient();
584  oCommandServer.StopAccept();
585  if( pCommandSocket == NULL )
586  {
587  std::cerr << "VA Server: Client applied, but did not connect to "
588  << "command socket on port [" << iCommandPort << "]" << std::endl;
589  return VA_CONNECTION_FAILED;
590 
591  }
592  oNewData.m_pCommandChannel = new VistaConnectionIP( pCommandSocket );
593  oNewData.m_pCommandChannel->SetIsBlocking( true );
594  oNewData.m_pCommandChannel->SetByteorderSwapFlag( VistaSerializingToolset::DOES_NOT_SWAP_MULTIBYTE_VALUES );
595 #ifdef VANET_SERVER_SHOW_RAW_TRAFFIC
596  oNewData.m_pCommandChannel->SetShowRawSendAndReceive( true );
597 #else
598  oNewData.m_pCommandChannel->SetShowRawSendAndReceive( false );
599 #endif
600 
601  // check for optional head connection
602  VANetCompat::sint32 iHeadChannelMode;
603  pTemporaryConnection->ReadInt32( iHeadChannelMode );
604  if( iHeadChannelMode == 0 )
605  {
606  // no separate head connection
607  oNewData.m_pHeadChannel = NULL; // only created upon client request
608  }
609  else if( iHeadChannelMode == 1 )
610  {
611  // separate TCP connection
612  VANetCompat::sint32 iHeadPort = GetNextFreePort();
613  VistaTCPServer oHeadServer( m_sServerIP, iHeadPort );
614  if( oCommandServer.GetIsValid() == false )
615  {
616  std::cerr << "VA Server: Client applied, but could not open "
617  << "head socket on port [" << iHeadPort << "]" << std::endl;
618  pTemporaryConnection->WriteInt32( -1 );
619  delete oNewData.m_pCommandChannel;
620  oNewData.m_pCommandChannel = NULL;
621  return VA_CONNECTION_FAILED;
622  }
623  pTemporaryConnection->WriteInt32( iHeadPort );
624  VistaTCPSocket* pHeadSocket = oHeadServer.GetNextClient();
625  oHeadServer.StopAccept();
626  if( pHeadSocket == NULL )
627  {
628  std::cerr << "VA Server: Client applied, but did not connect to "
629  << "head socket on port [" << iHeadPort << "]" << std::endl;
630  delete oNewData.m_pCommandChannel;
631  oNewData.m_pCommandChannel = NULL;
632  return VA_CONNECTION_FAILED;
633  }
634  oNewData.m_pHeadChannel = new VistaConnectionIP( pHeadSocket );
635  }
636  else if( iHeadChannelMode == 2 )
637  {
638  // separate UDP connection
639  VANetCompat::sint32 iHeadPort = GetNextFreePort();
640  oNewData.m_pHeadChannel = new VistaConnectionIP( m_sServerIP, iHeadPort );
641  if( oNewData.m_pHeadChannel->GetIsConnected() == false )
642  {
643  std::cerr << "VA Server: Client applied, but could not open "
644  << "UDP receiver on port [" << iHeadPort << "]" << std::endl;
645  pTemporaryConnection->WriteInt32( -1 );
646  delete oNewData.m_pHeadChannel;
647  oNewData.m_pHeadChannel = NULL;
648  delete oNewData.m_pCommandChannel;
649  oNewData.m_pCommandChannel = NULL;
650  return VA_CONNECTION_FAILED;
651  }
652  pTemporaryConnection->WriteInt32( iHeadPort );
653  }
654 
655  // Create Event Channel - while the client may not necessarily want to receive
656  // core events/progress updatzes, we still need this to pass other net-specific
657  // events like server close/reset
658 
659  VANetCompat::sint32 iEventPort = GetNextFreePort();
660  if( iEventPort == -1 )
661  {
662  std::cerr << "VA Server: Client applied, but no more free ports available" << std::endl;
663  }
664 
665  VistaTCPServer oEventServer( m_sServerIP, iEventPort );

783 void CVANetServerImpl::RemoveClient( int iClientID, int iStatusCode )
784 {
785  VistaMutexLock oServerLock( *m_pServerChangeLock );
786  VistaMutexLock oEventChannelLock( *m_pEventChannelLock );
787  VistaMutexLock oHandlerLock( *m_pEventHandlerLock );
788 
789  if( m_vecClients.size() == 0 )
790  return; // Clients already gone ...
791 
792  std::vector<CClientData>::iterator itClient = m_vecClients.begin() + iClientID;
793 
794 #ifdef VANET_SERVER_VERBOSE
795  std::cout << "VANetServer: Removing client [" << ( *itClient ).m_sHost << "]" << std::endl;
796 #endif
797 
798  if( iStatusCode != 0 && ( *itClient ).m_pEventChannel && ( *itClient ).m_pEventChannel->GetIsConnected() )
799  {
800  try
801  {
802  CVANetMessage oMessage( ( *itClient ).m_pEventChannel->GetByteorderSwapFlag() );
803  oMessage.SetConnection( ( *itClient ).m_pEventChannel );
804  oMessage.SetMessageType( CVANetNetworkProtocol::VA_EVENT_NET_EVENT );
805  oMessage.WriteInt( iStatusCode );
806  oMessage.WriteMessage();
807  ( *itClient ).m_pEventChannel->WaitForSendFinish();
808  }
809  catch( VistaExceptionBase& )
810  {
811  // nothing tho do, we're already cleaning up
812  }
813  catch( CVAException& )
814  {
815  // nothing tho do, we're already cleaning up
816  }
817  }
818 
819  // we assume that all invalidated connections have been closed, but not deleted before
820 
821  // first, we try to send the client the close command
822  if( ( *itClient ).m_pCommandChannel != NULL )
823  {
824  m_pConnObserver->RemoveObservedConnection( ( *itClient ).m_pCommandChannel );
825  m_pProtocol->HandleConnectionClose( ( *itClient ).m_pCommandChannel );
826  if( ( *itClient ).m_pCommandChannel->GetIsConnected()
827  && ( *itClient ).m_pCommandChannel->GetIsOpen() )
828  {
829  ( *itClient ).m_pCommandChannel->Close();
830  }
831  delete ( *itClient ).m_pCommandChannel;
832  }
833 
834  if( ( *itClient ).m_pHeadChannel != NULL )
835  {
836  m_pConnObserver->RemoveObservedConnection( ( *itClient ).m_pHeadChannel );
837  m_pConnObserver->RemoveObservedConnection( ( *itClient ).m_pHeadChannel );
838  if( ( *itClient ).m_pHeadChannel->GetIsConnected()
839  && ( *itClient ).m_pHeadChannel->GetIsOpen() )
840  {
841  ( *itClient ).m_pHeadChannel->Close();
842  }
843  delete ( *itClient ).m_pHeadChannel;
844  }
845 
846  if( ( *itClient ).m_pEventChannel != NULL )
847  {
848  if( m_pEventHandler )
849  m_pEventHandler->RemoveClientConnection( ( *itClient ).m_pEventChannel );
850  m_pConnObserver->RemoveObservedConnection( ( *itClient ).m_pEventChannel );
851  if( ( *itClient ).m_pEventChannel->GetIsConnected()
852  && ( *itClient ).m_pEventChannel->GetIsOpen() )
853  {
854  ( *itClient ).m_pEventChannel->Close();
855  }
856  delete ( *itClient ).m_pEventChannel;
857  }
858 
859  CEvent oEvent( CEvent::EVENT_CLIENT_DISCONNECT, ( *itClient ).m_sHost );
860  m_vecClients.erase( itClient );
861 
862  // to avoid deadlocks, we unlock before emitting the event
863  oServerLock.Unlock();
864  oHandlerLock.Unlock();
865  oEventChannelLock.Unlock();
866  EmitEvent( oEvent );
867 }
