69 const float* CVAEngineSignalSource::GetStreamBlock( const CVAAudiostreamState* )
70 {
71  m_sbBuffer.Zero();
72  double dOmega;
73  double dK = m_fK;
74 
75  std::map< double, double>::const_iterator cit = m_oConfig.lFreqModesSpectrum.begin();
76  while( cit != m_oConfig.lFreqModesSpectrum.end() )
77  {
78  const double& dBaseFrequency( cit->first );
79  const double& dAmplitude( cit->second );
80  const double dPeriodSamples = dK * GetSampleRate() / dBaseFrequency;
81  double& dPhase( m_lFreqModesPhase[dBaseFrequency] );
82  dOmega = ITAConstants::TWO_PI_F / dPeriodSamples; // 2*pi/T
83 
84  for( unsigned int i=0; i < GetBlocklength(); i++ )
85  {
86  m_sbBuffer[i] += float( dAmplitude * sin( dOmega * i + dPhase ) );
87  }
88 
89  dPhase = fmod( double( GetBlocklength() * dOmega + dPhase ), double( ITAConstants::TWO_PI_F ) );
90 
91  cit++;
92  }
93 
94  return m_sbBuffer.data();
95 }
