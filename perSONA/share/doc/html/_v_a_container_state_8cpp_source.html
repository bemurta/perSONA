175 void CVAContainerState::Diff( const CVAContainerState* pComp, std::vector<int>& viNew, std::vector<int>& viDel, std::vector<int>& viCom ) const
176 {
177  viNew.clear();
178  viDel.clear();
179  viCom.clear();
180 
181  // Vergleichszustand Leer => Alles Neu
182  if( pComp == nullptr )
183  {
184 
185  for( std::vector<ElemType>::const_iterator it = data.vElements.begin(); it != data.vElements.end(); ++it )
186  viNew.push_back( it->first );
187 
188  return;
189  }
190 
191  // Resort arrays when necessary
192  if( pComp->data.bDirty )
193  {
194  std::sort( pComp->data.vElements.begin(), pComp->data.vElements.end(), ElemCompare() );
195  pComp->data.bDirty = false;
196  }
197 
198  if( data.bDirty )
199  {
200  std::sort( data.vElements.begin(), data.vElements.end(), ElemCompare() );
201  data.bDirty = false;
202  }
203 
204  // Turbo vergleich
205  std::vector<ElemType>& A = pComp->data.vElements;
206  std::vector<ElemType>& B = this->data.vElements;
207  std::vector<ElemType>::iterator ai = A.begin();
208  std::vector<ElemType>::iterator bi = B.begin();
209 
210  while( true )
211  {
212  // Keine weiteren Elemente in A
213  if( ai == A.end() )
214  {
215  // Alle weiteren Elemente in B => neue Elemente
216  for( ; bi != B.end(); ++bi )
217  viNew.push_back( bi->first );
218  return;
219  }
220 
221  // Keine weiteren Elemente in B
222  if( bi == B.end() )
223  {
224  // Alle weiteren Elemente in A => gelöschte Elemente
225  // Alle weiteren Elemente in B => neue Elemente
226  for( ; ai != A.end(); ++ai )
227  viDel.push_back( ai->first );
228  return;
229  }
230 
231 
232  if( ai->first < bi->first )
233  {
234  // A Element fehlt in B
235  viDel.push_back( ai->first );
236 
237  // Weiter in A bis B erreicht
238  ai++;
239  continue;
240  }
241 
242  if( ai->first > bi->first )
243  {
244  // B Element fehlt in A
245  viNew.push_back( bi->first );
246 
247  // Weiter in B bis A erreicht
248  bi++;
249  continue;
250  }
251 
252  // *ai = *bi => Element in beiden und weiter
253  viCom.push_back( ai->first );
254  ai++;
255  bi++;
256  }
257 }
