45 int CVAObjectRegistry::RegisterObject( CVAObject* pObject )
46 {
47  if( !pObject )
48  VA_EXCEPT2( INVALID_PARAMETER, "Null pointer now allowed" );
49 
50  int iCurID = pObject->GetObjectID();
51  if( iCurID < 0 )
52  VA_EXCEPT2( INVALID_PARAMETER, "Object has an invalid ID" );
53 
54  // Ensure that it is not part of another registry
55  if( iCurID != 0 )
56  VA_EXCEPT2( INVALID_PARAMETER, "Object is part of another object registry" );
57 
58  std::string sName( pObject->GetObjectName() );
59  if( pObject->m_sObjectName.empty() )
60  VA_EXCEPT2( INVALID_PARAMETER, "Object must have a name" );
61  std::transform( sName.begin(), sName.end(), sName.begin(), toupper );
62 
63  // Ensure that the object has a unique name
64  if( m_mpObjects.find( sName ) != m_mpObjects.end() )
65  VA_EXCEPT2( INVALID_PARAMETER, "Object with name '" + sName + "' already registered as a VA object" );
66 
67  // Grow the internal lookup table if there is no free space left
68  if( m_nObjects == m_vpObjects.size() )
69  {
70  m_vpObjects.resize( m_vpObjects.size() + VA_OBJECTREGISTRY_GROW_SIZE, NULL );
71  }
72 
73  // Generate a new ID. Search until the end is reached. Then start again at the front.
74  int i = m_iIDLast, iID = 0;
75  while( iID == 0 )
76  {
77  i = ( i + 1 ) % ( int ) m_vpObjects.size();
78  if( m_vpObjects[ i ] == 0 )
79  {
80  iID = i;
81  break;
82  }
83  }
84 
85  pObject->SetObjectID( iID );
86  m_vpObjects[ iID ] = pObject;
87  m_mpObjects.insert( std::make_pair( sName, pObject ) );
88  m_iIDLast = iID;
89  ++m_nObjects;
90 
91  return iID;
92 }

94 void CVAObjectRegistry::UnregisterObject( CVAObject* pObject )
95 {
96  if( !pObject )
97  VA_EXCEPT2( INVALID_PARAMETER, "Null pointer now allowed" );
98 
99  int iCurID = pObject->GetObjectID();
100  if( iCurID <= 0 )
101  VA_EXCEPT2( INVALID_PARAMETER, "Object has an invalid ID" );
102 
103  if( iCurID >= ( int ) m_vpObjects.size() )
104  VA_EXCEPT2( INVALID_PARAMETER, "Object not registered" );
105 
106  if( m_vpObjects[ iCurID ] != pObject )
107  VA_EXCEPT2( INVALID_PARAMETER, "Object not registered" );
108 
109  m_vpObjects[ iCurID ] = NULL;
110  std::string sName( pObject->GetObjectName() );
111  std::transform( sName.begin(), sName.end(), sName.begin(), toupper );
112  m_mpObjects.erase( sName );
113  --m_nObjects;
114 }
