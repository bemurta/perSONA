189 void CVAHRIRDatasetSH::generateHankelLUT()
190 {
191  int orderIndex = 0;
192  int rIndex = 0;
193  int maxOrder = getMaxSHOrder();
194  dHankel = 0.01;
195  numPoints = 2/dHankel;
196 
197  double freqDist = samplingRate/2/(numFreq-1);
198  double k;
199  double kr;
200  ippDataType* hankel = 0;
201  hankelLUT.resize(numFreq);
202  std::complex<long double> referenceHankel;
203  std::complex<long double> tmpHankel, tmpHankel2;
204  for (int freqIndex = 1; freqIndex < numFreq; ++freqIndex)
205  {
206  hankelLUT[freqIndex].resize(maxOrder+1);
207  //calculate the frequency
208  k = 2*pi*freqIndex*freqDist/c;
209 
210  for (orderIndex = 0; orderIndex <= maxOrder; ++orderIndex)
211  {
212  hankelLUT[freqIndex][orderIndex].resize(numPoints);
213  //calculate the reference
214  referenceHankel.real(boost::math::sph_bessel(orderIndex,k*measurementDistance));
215  referenceHankel.imag(-boost::math::sph_neumann(orderIndex, k*measurementDistance));
216  for (rIndex = 1; rIndex <= numPoints; ++rIndex)
217  {
218  //calculate kr
219  kr = k*rIndex*dHankel;
220  hankel = myIppMalloc(1);
221  tmpHankel.real(boost::math::sph_bessel(orderIndex,kr));
222  tmpHankel.imag(-boost::math::sph_neumann(orderIndex, kr));
223  //divide hankel and referenceHankel
224  tmpHankel2 = tmpHankel/referenceHankel;
225  //tmpHankel2 = tmpHankel;
226 
227  hankel->re = tmpHankel2.real();
228  hankel->im = tmpHankel2.imag();
229  hankelLUT[freqIndex][orderIndex][rIndex-1] = hankel;
230  }
231  }
232  }
233 
234  maxIndexLUT.resize(maxOrder+1);
235  for (orderIndex = 0; orderIndex <= maxOrder; ++orderIndex)
236  {
237  maxIndexLUT[orderIndex] = pow((double)(orderIndex+1),2);
238  }
239 }

241 bool CVAHRIRDatasetSH::getUseHankel(double k,double r,int maxSHOrder)
242 {
243  // polyfit revealed that the 0.25 db error line between 1/r and hankel
244  // lies on y = (6.143*r-0.606)*maxOrder + 2.0843*r-0.3318
245  // so if kr > y for order x then don't use hankel (but 1/r)
246  if (k*r > (6.1435*r-0.606)*maxSHOrder+2.0843*r-0.3318)
247  {
248  return false;
249  }
250  return true;
251 }

609 void CVAHRIRDatasetSH::getHankelFromLUT(int order, int freqIndex, double r,ippDataType* data)
610 {
611  int minIndex = 0;
612  int maxIndex = 0;
613  int degreeIndex = -1;
614  ippDataType hankel;
615  double u = r/dHankel;
616  int rIndex(u);
617  u -= rIndex;
618  --rIndex;
619  if (rIndex >= numPoints-1)
620  {
621  rIndex = numPoints-2;
622  }
623  for (int orderIndex = 0; orderIndex <= order; ++orderIndex)
624  {
625  minIndex = maxIndex;
626  maxIndex = maxIndexLUT[orderIndex];
627  /* Linear RRW-Interpolation: very fast*/
628  //krIndex = std::floor(kr*2)-1;

804  double x = dRadius*sin(dElevation)*cos(dAzimuth);

808  //add the offset

810  x += header.centerX;
811  y += header.centerY;
812  z += header.centerZ;
813 
814  //transform back
815  dRadius = sqrt(x*x+y*y+z*z);
816  dElevation = std::acos(z/dRadius);
817  dAzimuth = std::atan2(y,x);
818 
819 
820  //transform the measurement distance
821  x = dMeasurementDistance*sin(dElevation)*cos(dAzimuth);
822  y = dMeasurementDistance*sin(dElevation)*sin(dAzimuth);
823  z = dMeasurementDistance*cos(dElevation);

827  x += header.centerX;
828  y += header.centerY;
829  z += header.centerZ;
830 
831  //transform back
