189 CVASceneState* CVASceneManager::CreateDerivedSceneState( const int iBaseStateID, const double dModificationTime )
190 {
191  ITACriticalSectionLock oLock( m_csStates );
192 
193  SSMap::const_iterator cit = m_mSceneStates.find( iBaseStateID );
194  assert( cit != m_mSceneStates.end() );
195  const CVASceneState* pBaseState = cit->second;
196 
197  // Base state must be fixated.
198  assert( pBaseState->IsFixed() );
199 
200  // Derive
201  const int iNewStateID = m_iIDCounterScene++;
202  CVASceneState* pNewState = dynamic_cast< CVASceneState* >( m_pPoolSceneStates->RequestObject() );
203  pNewState->SetManager( this );
204  pNewState->Copy( iNewStateID, dModificationTime, pBaseState );
205 
206  // Speichere diese (temporär)
207  m_mSceneStates.insert( SSMapItem( iNewStateID, pNewState ) );
208 
209  return pNewState;
210 }
