216 void CVAAmbisonicsBinauralMixdownReproduction::UpdateScene( CVASceneState* pNewState )
217 {
218  if( m_iListenerID == -1 )
219  return;
220 
221  CVAListenerState* pLIstenerState( pNewState->GetListenerState( m_iListenerID ) );
222  if( pLIstenerState == nullptr )
223  return;
224 
225  const CVADirectivityDAFFHRIR* pHRIRSet = ( CVADirectivityDAFFHRIR*) pLIstenerState->GetDirectivity();
226 
227  if( pHRIRSet == nullptr )
228  return;
229 
230  VAVec3 vListenerPos, vListenerView, vListenerUp;
231  m_oParams.pCore->GetSoundReceiverRealWorldPositionOrientationVU( m_iListenerID, vListenerPos, vListenerView, vListenerUp );
232 
233  for( int i=0; i<GetNumVirtualLoudspeaker(); i++ )
234  {
235  int iIndexLeft = 2*i+0;
236  int iIndexRight = 2*i+1;
237 
238  const CVAHardwareDevice* pDevice( m_pVirtualOutput->vpDevices[i] );
239 
240  double dAzimuth = GetAzimuthOnTarget_DEG( vListenerPos, vListenerView, vListenerUp, pDevice->vPos );
241  double dElevation = GetElevationOnTarget_DEG( vListenerPos, vListenerView, vListenerUp, pDevice->vPos );
242 
243  double dDistance = ( pDevice->vPos - vListenerPos ).Length();
244 
245  // Re-init temp buffer to match with current HRIR length
246  int iHRIRFilterLength = pHRIRSet->GetProperties()->iFilterLength;
247  if( m_sfHRIRTemp.length() != iHRIRFilterLength )
248  m_sfHRIRTemp.init( 2, iHRIRFilterLength, false );
249 
250  int iIndex;
251  pHRIRSet->GetNearestNeighbour( float( dAzimuth ), float( dElevation ), &iIndex );
252 
253  // Skip if still same HRIR
254  if( iIndex == m_viLastHRIRIndex[i] )
255  continue;
256 
257  pHRIRSet->GetHRIRByIndex( &m_sfHRIRTemp, iIndex, 0 );
258  m_viLastHRIRIndex[i] = iIndex;
259

261  ITAUPFilter* pFilterChR = m_pdsStreamFilter->pFilterPool->RequestFilter();
262 
263  int iLoadFilter = (std::min)( m_iHRIRFilterLength, iHRIRFilterLength );
264  pFilterChL->Zeros();
265  pFilterChL->Load( m_sfHRIRTemp[0].data(), iLoadFilter );
266  pFilterChR->Zeros();
267  pFilterChR->Load( m_sfHRIRTemp[1].data(), iLoadFilter );
268 
269  // Update
270  m_pdsStreamFilter->vpConvolver[iIndexLeft]->ExchangeFilter( pFilterChL );
271  m_pdsStreamFilter->vpConvolver[iIndexRight]->ExchangeFilter( pFilterChR );
272 
273  pFilterChL->Release();
274  pFilterChR->Release();
275  }
276 
277  m_pdsStreamFilter->pTrigger->trigger();
278 
279  return;
280 }
