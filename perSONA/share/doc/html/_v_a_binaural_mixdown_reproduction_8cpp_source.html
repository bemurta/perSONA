130 void CVABinauralMixdownReproduction::UpdateScene( CVASceneState* pNewState )
131 {
132  if( m_iListenerID == -1 )
133  return;
134 
135  CVAListenerState* pLIstenerState( pNewState->GetListenerState( m_iListenerID ) );
136  if( pLIstenerState == nullptr )
137  return;
138 
139  const CVADirectivityDAFFHRIR* pHRIRSet = ( CVADirectivityDAFFHRIR *) pLIstenerState->GetDirectivity();
140 
141  if( pHRIRSet == nullptr )
142  return;
143 
144  VAVec3 vListenerPos, vListenerView, vListenerUp;
145  m_oParams.pCore->GetSoundReceiverRealWorldPositionOrientationVU( m_iListenerID, vListenerPos, vListenerView, vListenerUp );
146 
147  for( int i=0; i<GetNumVirtualLoudspeaker(); i++ )
148  {
149  int iIndexLeft = 2*i+0;
150  int iIndexRight = 2*i+1;
151 
152  const CVAHardwareDevice* pDevice( m_pVirtualOutput->vpDevices[i] );
153 
154  double dAzimuth = GetAzimuthOnTarget_DEG( vListenerPos, vListenerView, vListenerUp, pDevice->vPos );
155  double dElevation = GetElevationOnTarget_DEG( vListenerPos, vListenerView, vListenerUp, pDevice->vPos );
156 
157  double dDistance = ( pDevice->vPos - vListenerPos ).Length();
158 
159  // Re-init temp buffer to match with current HRIR length
160  int iHRIRFilterLength = pHRIRSet->GetProperties()->iFilterLength;
161  if( m_sfHRIRTemp.length() != iHRIRFilterLength )
162  {
163  m_sfHRIRTemp.init( 2, iHRIRFilterLength, false );
164  }
165  int iIndex;
166  pHRIRSet->GetNearestNeighbour( float( dAzimuth ), float( dElevation ), &iIndex );
167 
168  // Skip if still same HRIR
169  if( iIndex == m_viLastHRIRIndex[i] )
170  continue;
171 
172  pHRIRSet->GetHRIRByIndex( &m_sfHRIRTemp, iIndex, 0 );
173  m_viLastHRIRIndex[i] = iIndex;
174 
175  ITAUPFilter* pFilterChL = m_pdsStreamFilter->pFilterPool->RequestFilter();
176  ITAUPFilter* pFilterChR = m_pdsStreamFilter->pFilterPool->RequestFilter();
177 
178  int iLoadFilter = (std::min)( m_iHRIRFilterLength, iHRIRFilterLength );
179  pFilterChL->Zeros();
180  pFilterChL->Load( m_sfHRIRTemp[0].data(), iLoadFilter );
181  pFilterChR->Zeros();
182  pFilterChR->Load( m_sfHRIRTemp[1].data(), iLoadFilter );
183 
184  // Update
185  m_pdsStreamFilter->vpConvolver[iIndexLeft]->ExchangeFilter( pFilterChL );
186  m_pdsStreamFilter->vpConvolver[iIndexRight]->ExchangeFilter( pFilterChR );
187 
188  pFilterChL->Release();
189  pFilterChR->Release();
190  }
191 
192  m_pdsStreamFilter->pTrigger->trigger();
193 
194  return;
195 }

204 StreamFilter::StreamFilter( int iNumChannels, double dSampleRate, int iBlockLength, int iMaxFilterLength )
205 : ITADatasourceRealization( 2, dSampleRate, iBlockLength )
206 , pdsInput( NULL )
207 , pFilterPool( NULL )
208 , pTrigger( NULL )
209 {
210  pFilterPool = new ITAUPFilterPool( (int) m_uiBlocklength, iMaxFilterLength, 24 );
211  pDirac = pFilterPool->RequestFilter();
212  pDirac->identity();
213 
214  pTrigger = new ITAUPTrigger;
215 
216  for( int i=0; i<2*iNumChannels; i++ )
217  {
218  ITAUPConvolution* pConvolver = new ITAUPConvolution( (int) GetBlocklength(), iMaxFilterLength, pFilterPool );
219  pConvolver->SetFilterExchangeTrigger( pTrigger );
220  pConvolver->SetFilterExchangeMode( ITAUPConvolution::CROSSFADE_COSINE_SQUARE );
221  pConvolver->SetFilterCrossfadeLength( (int) GetBlocklength() );
222  pConvolver->ExchangeFilter( pDirac );
223 
224  vpConvolver.push_back( pConvolver );
225  }
226 
227  pTrigger->trigger();
228 }

239 void StreamFilter::ProcessStream( const ITAStreamInfo* pStreamInfo )
240 {
241  float* pfBinauralOutputDataChL = GetWritePointer( 0 );
242  float* pfBinauralOutputDataChR = GetWritePointer( 1 );
243 
244  for( unsigned int i=0; i<pdsInput->GetNumberOfChannels(); i++ )
245  {
246  int iIndexLeft = 2*i+0;
247  int iIndexRight = 2*i+1;
248  const float* pfInputData = pdsInput->GetBlockPointer( i, pStreamInfo );
249  int iWriteMode = ( i == 0 ) ? ITAUPConvolution::OUTPUT_OVERWRITE : ITAUPConvolution::OUTPUT_MIX;
250  vpConvolver[iIndexLeft]->Process( pfInputData, pfBinauralOutputDataChL, iWriteMode );
251  vpConvolver[iIndexRight]->Process( pfInputData, pfBinauralOutputDataChR, iWriteMode );
252  }
253 
254  IncrementWritePointer();
255 }
