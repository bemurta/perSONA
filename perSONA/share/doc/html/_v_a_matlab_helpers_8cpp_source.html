421 mxArray* matlabCreateStruct( const CVAStruct& oStruct )
422 {
423  std::vector< const char* > vcpFieldNames;
424  CVAStruct::const_iterator it = oStruct.Begin();
425  while( it != oStruct.End() )
426  {
427  const std::string sKey( ( *it ).first );
428  char* cFieldName = _strdup( sKey.c_str() );
429  vcpFieldNames.push_back( cFieldName );
430  it++;
431  }
432 
433  const size_t nDims = 1;
434  const int nFields = int( vcpFieldNames.size() );
435  const char** ppszFieldNames = &( *vcpFieldNames.begin() );
436  mxArray* pStruct = mxCreateStructArray( 1, &nDims, nFields, ppszFieldNames );
437 
438  for( size_t i = 0; i < nFields; i++ )
439  {
440  const std::string sFieldName( vcpFieldNames[ i ] );
441  assert( oStruct.HasKey( sFieldName ) );
442 
443  const CVAStructValue* pStructValue( oStruct.GetValue( sFieldName ) );
444 
445  mxArray* p;
446  int iType = pStructValue->GetDatatype();
447  switch( iType )
448  {
449  case CVAStructValue::BOOL:
450  {
451  mxSetField( pStruct, 0, sFieldName.c_str(), mxCreateLogicalScalar( *pStructValue ) );
452  break;
453  }
454  case CVAStructValue::INT:
455  {
456  size_t iVal = 1;
457  p = mxCreateNumericArray( 1, &iVal, mxINT32_CLASS, mxREAL );
458  *( ( int* ) mxGetData( p ) ) = *pStructValue;
459  mxSetField( pStruct, 0, sFieldName.c_str(), p );
460  break;
461  }
462  case CVAStructValue::DOUBLE:
463  {
464  mxSetField( pStruct, 0, sFieldName.c_str(), mxCreateDoubleScalar( *pStructValue ) );
465  break;
466  }
467  case CVAStructValue::STRING:
468  {
469  std::string s = *pStructValue;
470  mxSetField( pStruct, 0, sFieldName.c_str(), mxCreateString( s.c_str() ) );
471  break;
472  }
473  case CVAStructValue::STRUCT:
474  {
475  mxSetField( pStruct, 0, sFieldName.c_str(), matlabCreateStruct( *pStructValue ) );
476  break;
477  }
478  case CVAStructValue::DATA:
479  {
480  int nSamples = int( pStructValue->GetDataSize() / 4 );
481  size_t nDims[ 2 ] = { 1, nSamples };
482  p = mxCreateNumericArray( 2, nDims, mxSINGLE_CLASS, mxREAL );
483  mxSetField( pStruct, 0, sFieldName.c_str(), p );
484  const float* pfSrcData = ( const float* ) ( pStructValue->GetData() );
485  float* pfDestData = ( float* ) mxGetPr( p );
486  for( size_t n = 0; n < nSamples; n++ )
487  pfDestData[ n ] = pfSrcData[ n ];
488 
489  break;
490  }
491  default:
492  {
493  VA_EXCEPT2( INVALID_PARAMETER, "Could not read VAStruct value at key '" + sFieldName + "'" );
494  }
495  }
496 
497  delete vcpFieldNames[ i ];
498  }
499 
500  return pStruct;
501 }
