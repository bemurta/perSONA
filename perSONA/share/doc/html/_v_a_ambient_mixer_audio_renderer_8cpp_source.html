65 void CVAAmbientMixerAudioRenderer::HandleProcessStream( ITADatasourceRealization*, const ITAStreamInfo* )
66 {
67  if( m_bIndicateReset )
68  {
69  m_bResetAck = true;
70  m_bIndicateReset = false;
71  }
72 
73  if( m_pNewSceneState )
74  {
75  if( m_pCurSceneState )
76  m_pCurSceneState->RemoveReference();
77  m_pCurSceneState = m_pNewSceneState;
78  m_pNewSceneState = NULL;
79  }
80  else
81  {
82  if( m_pCurSceneState == nullptr )
83  return;
84  }
85 
86  for( int n=0; n < int( m_pDataSource->GetNumberOfChannels() ); n++ )
87  {
88  float* pfOutBuf = m_pDataSource->GetWritePointer( (unsigned int)( n ) );
89  fm_zero( pfOutBuf, m_pDataSource->GetBlocklength() );
90  }
91 
92  std::vector< int > viSoundSourceIDs;
93  m_pCurSceneState->GetSoundSourceIDs( &viSoundSourceIDs );
94 
95  for( size_t i=0; i < viSoundSourceIDs.size(); i++ )
96  {
97  int iID = viSoundSourceIDs[i];
98  CVASoundSourceDesc* pDesc = m_oParams.pCore->GetSceneManager()->GetSoundSourceDesc( iID );
99  CVASoundSourceState* pState = m_pCurSceneState->GetSoundSourceState( iID );
100 
101  assert( pDesc && pState );
102 
103  bool bForThisRenderer = ( pDesc->sExplicitRendererID == m_oParams.sID || pDesc->sExplicitRendererID.empty() );
104  if( !bForThisRenderer || pDesc->bMuted )
105  continue;
106 
107  const ITASampleBuffer* psbInput = pDesc->pSignalSourceInputBuf;
108  double dGainFactor = pState->GetVolume();
109 
110  // Add samples to out buffer
111  for( int n=0; n < int( m_pDataSource->GetNumberOfChannels() ); n++ )
112  {
113  float* pfOutBuf = m_pDataSource->GetWritePointer( (unsigned int)( n ) );
114  for( int j=0; j< int( m_pDataSource->GetBlocklength() ); j++ )
115  pfOutBuf[j] += psbInput->data()[j] * float( dGainFactor );
116  }
117  }
118  m_pDataSource->IncrementWritePointer();
119 }
