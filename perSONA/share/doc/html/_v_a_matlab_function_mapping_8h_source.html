142 inline mxArray* CreateFunctionInputArgumentStruct( std::vector<FunctionInputArgument>& vInputArgs )
143 {
144  const mwSize nFields = 5;
145  const char* ppszFieldNames[] = { "name", "type", "desc", "optional", "default" };
146 
147  mxArray* pStruct = mxCreateStructMatrix( 1, int( vInputArgs.size() ), nFields, ppszFieldNames );
148  for( size_t i = 0; i < vInputArgs.size(); i++ ) {
149  mxSetField( pStruct, ( mwIndex ) i, ppszFieldNames[ 0 ], mxCreateString( vInputArgs[ i ].sName.c_str() ) );
150  mxSetField( pStruct, ( mwIndex ) i, ppszFieldNames[ 1 ], mxCreateString( vInputArgs[ i ].sType.c_str() ) );
151  mxSetField( pStruct, ( mwIndex ) i, ppszFieldNames[ 2 ], mxCreateString( vInputArgs[ i ].sDesc.c_str() ) );
152  mxSetField( pStruct, ( mwIndex ) i, ppszFieldNames[ 3 ], mxCreateLogicalScalar( vInputArgs[ i ].bOptional ) );
153  mxSetField( pStruct, ( mwIndex ) i, ppszFieldNames[ 4 ], mxCreateString( vInputArgs[ i ].sDefaultValue.c_str() ) );
154  }
155 
156  return pStruct;
157 };
