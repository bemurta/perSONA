119 void CVANetMessage::ReadMessage()
120 {
121  try
122  {
123  VANetCompat::sint32 nMessageSize;
124  int nReturn = m_pConnection->ReadInt32( nMessageSize );
125 
126  if( nReturn != sizeof( VANetCompat::sint32 ) )
127  VA_EXCEPT2( NETWORK_ERROR, "Received less bytes than expected" );
128 
129  // we need at least the three protocol ints
130  assert( nMessageSize >= 3 * sizeof( VANetCompat::sint32 ) );
131 
132  if( nMessageSize > ( int ) m_vecIncomingBuffer.size() )
133  m_vecIncomingBuffer.resize( nMessageSize );
134 
135  int iBytesReadTotal = 0;
136  while( nMessageSize != iBytesReadTotal )
137  {
138  int iIncommingBytes = m_pConnection->WaitForIncomingData( 0 );
139  int iBytesRead = 0;
140  if( iBytesReadTotal + iIncommingBytes > nMessageSize )
141  iBytesRead = m_pConnection->ReadRawBuffer( &m_vecIncomingBuffer[ iBytesReadTotal ], nMessageSize - iBytesReadTotal ); // read residual bytes
142  else
143  iBytesRead = m_pConnection->ReadRawBuffer( &m_vecIncomingBuffer[ iBytesReadTotal ], iIncommingBytes ); // read all incoming bytes
144 
145  iBytesReadTotal += iBytesRead;
146  }
147  assert( iBytesReadTotal == nMessageSize );
148 
149  m_oIncoming.SetBuffer( ( VistaType::byte* ) &m_vecIncomingBuffer[ 0 ], nMessageSize, false );
150  }
151  catch( VistaExceptionBase& ex )
152  {
153  VA_EXCEPT2( NETWORK_ERROR, ex.GetExceptionText() );
154  }
155 
156  m_nMessageType = ReadInt();
157  m_nExceptionMode = ReadInt();
158  m_nMessageId = ReadInt();
159 }

227

236  if( nMessageSize > ( int ) m_vecIncomingBuffer.size() )
237  m_vecIncomingBuffer.resize( nMessageSize );
238 
239  int iBytesReceivedTotal = 0;
240  while( nMessageSize > iBytesReceivedTotal )
241  {
242  int iIncommingBytes = m_pConnection->WaitForIncomingData( 0 );
243  int iBytesReceived = m_pConnection->ReadRawBuffer( &m_vecIncomingBuffer[ iBytesReceivedTotal ], iIncommingBytes );
244  iBytesReceivedTotal += iBytesReceived;

246

637 {

643  {

646

657

661
