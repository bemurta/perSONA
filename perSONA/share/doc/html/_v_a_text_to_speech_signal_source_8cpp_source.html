386 void CVATextToSpeechSignalSource::VisemeProcessing(CPRC_abuf * abuf, void * userdata) {
387 #ifdef VACORE_WITH_TTS_SIGNAL_SOURCE
388 
389 
390  //this callback is called per sentence in the text, so we need to append to the other visemes and also time-wise!
391  UserCallbackData* data = (UserCallbackData*)userdata;
392  float endTime = 0.0f;
393 
394  if (abuf == NULL){
395  VA_WARN("TextToSpeechSignalSource", "The buffer is NULL, cannot extract visemes!");
396  return;
397  }
398 
399  if (userdata == NULL){
400  VA_WARN("TextToSpeechSignalSource", "The userdata viseme string is NULL, cannot extract visemes!");
401  return;
402  }
403 
404  /* Transcriptions contain markers, phonetic information, a
405  list of these items is available for each audio buffer. */
406  const CPRC_abuf_trans * trans;
407  std::string label;
408  float start, end;
409  /* Process the transcription buffer items and print information. */
410  for (int i = 0; i < CPRC_abuf_trans_sz(abuf); i++) {
411  trans = CPRC_abuf_get_trans(abuf, i);
412  start = CPRC_abuf_trans_start(trans); /* Start time in seconds */
413  end = CPRC_abuf_trans_end(trans); /* End time in seconds */
414  label = CPRC_abuf_trans_name(trans); /* Label, type dependent */
415  if (CPRC_abuf_trans_type(trans) == CPRC_ABUF_TRANS_PHONE) {
416  //VA_INFO("TextToSpeechSignalSource", "Phoneme: " + std::to_string(start) + " " + std::to_string(end) + " " + label);
417  //visemes_time->first.append( "\t<viseme start=\"" + to_string_with_precision(start+visemes_time->second) + "\" articulation=\"1\" type=\"" + PhonemeToViseme(label) + "\" />\n");
418  std::string viseme = std::string("\t<lips ")
419  + "viseme=\"" + TTSEngine::getInstance().PhonemeToViseme(label) + "\" "
420  + "articulation=\"1.0\" "
421  + "start=\"" + to_string_with_precision(start + data->lastEnd) + "\" "
422  + "ready=\"" + to_string_with_precision(start + data->lastEnd) + "\" "
423  + "relax=\"" + to_string_with_precision(end + data->lastEnd) + "\" "
424  + "end=\"" + to_string_with_precision(end + data->lastEnd) + "\" "
425  + "/>\n";
426  data->visemes.append(viseme);
427  endTime = end;
428  }
429  /*else if (CPRC_abuf_trans_type(trans) == CPRC_ABUF_TRANS_WORD) {
430  VA_INFO("TextToSpeechSignalSource", "Word: " + std::to_string(start) + " " + std::to_string(end) + " " + label);
431  }
432  else if (CPRC_abuf_trans_type(trans) == CPRC_ABUF_TRANS_MARK) {
433  VA_INFO("TextToSpeechSignalSource", "Marker: " + std::to_string(start) + " " + std::to_string(end) + " " + label);
434  }*/
435  else if (CPRC_abuf_trans_type(trans) == CPRC_ABUF_TRANS_ERROR) {
436  VA_INFO("TextToSpeechSignalSource", "ERROR: could not retrieve transcription at "+ std::to_string(i));
437  }
438  }
439  data->lastEnd += endTime;
440 
441  //we basically append the data in abuf to the data in floatBuffer and on the way convert it from short to float (since that's what we need for the SoundSignal)
442  std::size_t bufferLength = CPRC_abuf_wav_sz(abuf);
443  std::size_t floatBufLength = data->floatBuffer.size();
444  data->floatBuffer.resize(bufferLength+data->floatBuffer.size());
445  stc_sint16_to_float(&data->floatBuffer[floatBufLength], CPRC_abuf_wav_data(abuf), bufferLength);
446 
447 #endif
