182 double GetAzimuthOnTarget_DEG( const VAVec3& vOriginPos, const VAVec3& vView, const VAVec3& vUp, const VAVec3& vTargetPos )
183 {
184  VAVec3 vDir = vTargetPos - vOriginPos;
185  vDir.Norm();
186  VAVec3 vViewMinusZ = vView * ( -1.0f ); // local z axis
187  const VAVec3 vRight = vViewMinusZ.Cross( vUp ); // local x axis
188  const double dAzimuthAngleDeg = atan2( vDir.Dot( vRight ), vDir.Dot( vView ) ) * 180.0f / ITAConstants::PI_D;
189  return ( ( dAzimuthAngleDeg < 0.0f ) ? ( dAzimuthAngleDeg + 360.0f ) : dAzimuthAngleDeg );
190 }
