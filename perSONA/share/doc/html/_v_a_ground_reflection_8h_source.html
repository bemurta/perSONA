38 bool calcGroundReflection(const RG_Vector& vSourcePos,
39  const RG_Vector& vListenerPos,
40  const float fGroundLevel, // Höhe der Bodenebene [m]
41  float& fPathLength, // Länge des Reflektionspfades [m]
42  float& fReflectionAngleRAD) // (Einfalls-/Ausfalls-)winkel [Bodenmaß]
43 {
44  /*
45  * Ansatz: - 3D => 2D konvertieren. Hörer stets im Nullpunkt.
46  * - Spekulare Reflektion an der Bodenebene (plan!)
47  * - Berechnung des Reflektionspunktes (2D)
48  * - Bestimmung des Umweges (Pfadlänge)
49  *
50  * Herleitung: [fwe] Blatt Papier + Maple ;-)
51  */
52 
53  // Abstand Quelle<->Hörer bei Projektion auf den Boden
54  float t1 = vSourcePos.x-vListenerPos.x;
55  float t2 = vSourcePos.z-vListenerPos.z;
56  float sx = sqrt( t1*t1 + t2*t2 );
57 
58  // Höhen Quelle, Hörer relativ zum Bodenniveau
59  float sy = vSourcePos.y - fGroundLevel;
60  float ly = vListenerPos.y - fGroundLevel;
61 
62  if ((sy < 0) || (ly <= 0)) return false;
63 
64  //assert( sy+ly > 0 ); // Klären was passieren soll wenn sich Hörer und Quelle auf einer Ebene mit dem Bodenniveau befinden
65  if ((sy+ly) == 0.0f) return false;
66 
67  // Entfernung des Reflektionspunktes vom Hörer
68  // Der Punkt liegt natürlich in der Bodenebene yr=0
69  float xr = (sx*ly)/(sy+ly);
70 
71  // Pfadlänge
72  t1 = (sx-xr);
73  t2 = xr;
74  fPathLength = sqrt( t1*t1 + sy*sy ) + sqrt( t2*t2 + ly*ly );
75 
76  // Reflektionswinkel
77  fReflectionAngleRAD = atan2( sx, sy+ly );
78 
79  return true;
80 }

82 // Winkelabhängiger Absorptionsgrad als Funktion des
83 // Wandnormalen Absorptiongrades
84 float calcAngleDependentAbsorptionCoeff(const float alpha_norm,
85  const float phiRAD)
86 {
87  /*
88  * Herleitung wiefolgt:
89  * - R_norm = sqrt(1-alpha_norm)
90  * - Daraus Z_norm = Z0*(1+R)/(1-R)
91  * - Dann hieraus R_phi = [Z_norm*cos(phi)-Z0] / [Z_norm*cos(phi)+Z0]
92  * - Dies dann wieder zu alpha_phi = 1-R_phi^2
93  *
94  * Man nehme Maple und koche diese Suppe durch... et voila:
95  */
96 
97  float cp = cos(phiRAD);
98  float alpha_phi = alpha_norm*cp / (4*(cp*(1+sqrt(1-alpha_norm))+cp+1));
99  return alpha_phi;
100 }
