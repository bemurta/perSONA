353 static int lSetTimer( lua_State *L )
354 {
355  double dPeriod = luaL_checknumber( L, 1 );
356  if( dPeriod <= 0 )
357  return luaL_error( L, "Timer period must be greater zero" );

359  // Neuen Timer erzeugen, falls noch nicht geschehen
